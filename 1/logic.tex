\lsection{記号論理学}

本節では、命題と呼ばれる対象と、それらの間の推論関係 $\vdash$ を公理的に扱う。

本節の各系の証明は、添付のLeanファイル\nolinkurl{math/1/logic.lean}を参照されたい。

\rem{古典論理}{
  本ノートでは、古典論理のみを扱う。
}

\rem{命題}{
  命題は、真偽に定めうる言明と考えてよい。
}


\lsubsection{命題結合子と推論規則}

\dfn{推論}{
  命題$\phi, \psi$について、$\phi$から$\psi$が推論されることを、$\phi \vdash \psi$で表す。

  このとき、$\phi$を前提、$\psi$を帰結と呼ぶ。
}

\rem{前提}{
  本ノートでは、文脈集合を連言で結合した命題を前提として扱う。
  これは一般的でない取り扱いであることに注意されたい。
}

\rem{命題論理の推論体系}{
  ここでは推論を行う上で許される規則として、
  後述する\axiref{同一律}、\axiref{カット規則}、\axiref{連言の導入則}、\axiref{連言の除去則}、\axiref{演繹定理}、\axiref{Modus Ponens}、\axiref{二重否定の除去}を与える。
}

\rem{命題の結合順序}{
  \num{1}つまたは\num{2}つの命題から新たな命題を作り出す記号は、命題結合子と呼ばれる。

  複数の命題結合子がある場合には結合される順序により与えられる命題が変わるため、厳密さのために$()$によって結合順序を定める。
}

\dfn{矛盾}{
  矛盾は命題である。
  これを$\bot$で表す。
}

\rem{矛盾}{
  矛盾は常に偽を与える命題と考えてよい。
}

\axi{同一律}{
  命題$\phi$について、以下を定める。
  \eq*{
    \phi \vdash \phi
  }
}

\axi{カット規則}{
  命題$\phi, \psi, \chi$について、以下を定める。
  \eq*{
    \phi \vdash \psi \metaand \psi \vdash \chi \metaimply \phi \vdash \chi
  }
}

\dfn{連言}{
  命題$\phi, \psi$について、$\phi \land \psi$は命題である。
  「$\phi$かつ$\psi$」と呼ぶ。
}

\axi{連言の導入則}{
  命題$\phi, \psi, \chi$について、以下を定める。
  \eq*{
    \phi \vdash \psi \metaand \phi \vdash \chi \metaimply \phi \vdash \psi \land \chi
  }
}

\axi{連言の除去則}{
  命題$\phi, \psi$について、以下の\num{2}つを定める。
  \eqg*{
    \phi \land \psi \vdash \phi \\*
    \phi \land \psi \vdash \psi
  }
}

\dfn{含意}{
  命題$\phi, \psi$について、$\phi \rightarrow \psi$は命題である。
  「$\phi$ならば$\psi$」と呼ぶ。
}

\axi{演繹定理}{
  命題$\phi, \psi, \chi$について、以下を定める。
  \eq*{
    \phi \land \psi \vdash \chi \metaimply \phi \vdash \psi \rightarrow \chi
  }
}

\axi{Modus Ponens}{
  命題$\phi, \psi$について、以下を定める。
  \eq*{
    \phi \land \qty(\phi \rightarrow \psi) \vdash \psi
  }
}

\axi{二重否定の除去}{
  命題$\phi$について、以下を定める。
  \eq*{
    \qty(\phi \rightarrow \bot) \rightarrow \bot \vdash \phi
  }
}


\lsubsection{いくつかの重要な定理}

\cor{弱化}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \psi \metaimply \phi \land \chi \vdash \psi
  }
}

\cor{連言の冪等律}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \phi \land \phi
  }
}

\cor{連言の結合律}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eqg*{
    \qty(\phi \land \psi) \land \chi \vdash \phi \land \qty(\psi \land \chi) \\*
    \phi \land \qty(\psi \land \chi) \vdash \qty(\phi \land \psi) \land \chi
  }
}

\cor{連言の交換律}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \phi \land \psi \vdash \psi \land \phi
  }
}

\cor{仮言三段論法}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eq*{
    \qty(\phi \rightarrow \psi) \land \qty(\psi \rightarrow \chi) \vdash \phi \rightarrow \chi
  }
}

\cor{逆演繹定理}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \psi \rightarrow \chi \metaimply \phi \land \psi \vdash \chi
  }
}

\corf{{\L}ukasiewiczの第一公理}{Lukasiewiczの第一公理}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \psi \rightarrow \phi
  }
}

\cor{無矛盾律}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi \land \qty(\phi \rightarrow \bot) \vdash \bot
  }
}

\cor{二重否定導入}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \qty(\phi \rightarrow \bot) \rightarrow \bot
  }
}

\cor{爆発律}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \bot \vdash \phi
  }
}

\cor{否定含意からの前件の取り出し}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \qty(\phi \rightarrow \psi) \rightarrow \bot \vdash \phi
  }
}

\cor{Peirce則}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \qty(\phi \rightarrow \psi) \rightarrow \phi \vdash \phi
  }
}

\cor{背理法}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \phi \land \qty(\psi \rightarrow \bot) \vdash \bot \metaimply \phi \vdash \psi
  }
}

\cor{対偶法}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eq*{
    \phi \rightarrow \bot \vdash \psi \rightarrow \bot \metaimply \psi \vdash \phi
  }
}


\lsubsection{同値}

\rem{命題の定義記号}{
  定義記号$\defiff$を導入する。

  これは左辺を用いて表現された命題は、議論において全てその左辺と一致する部分を右辺に置き換えて理解するという意味である。
}

\dfn{同値}{
  新たな命題結合子$\leftrightarrow$を定める。

  命題$\phi, \psi$について、命題$\phi \leftrightarrow \psi$を以下で定める。
  \eq*{
    \phi \leftrightarrow \psi \defiff \qty(\phi \rightarrow \psi) \land \qty(\psi \rightarrow \phi)
  }

  「$\phi$と$\psi$は必要十分」と呼ぶ。
}

\cor{連言における命題の代入原理}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eq*{
    \qty(\phi \land \chi) \land \qty(\phi \leftrightarrow \psi) \vdash \psi \land \chi
  }
}

\cor{含意における命題の代入原理}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eqg*{
    \qty(\phi \rightarrow \chi) \land \qty(\phi \leftrightarrow \psi) \vdash \psi \rightarrow \chi \\*
    \qty(\chi \rightarrow \phi) \land \qty(\phi \leftrightarrow \psi) \vdash \chi \rightarrow \psi
  }
}


\lsubsection{いくつかの命題結合子と重要な定理}

\dfn{真理}{
  新たな命題$\top$を以下で定める。
  \eq*{
    \top \defiff \bot \rightarrow \bot
  }
}

\rem{真理からの帰結}{
  命題$\phi$について、簡単のために$\top \vdash \phi$を$\vdash \phi$で表す。
}

\dfn{左含意}{
  新たな命題結合子$\leftarrow$を定める。

  命題$\phi, \psi$について、命題$\phi \leftarrow \psi$を以下で定める。
  \eq*{
    \phi \leftarrow \psi \defiff \psi \rightarrow \phi
  }
}

\dfn{否定}{
  新たな命題結合子$\lnot$を定める。

  命題$\phi$について、命題$\lnot \phi$を以下で定める。
  \eq*{
    \lnot \phi \defiff \phi \rightarrow \bot
  }
}

\dfn{選言}{
  新たな命題結合子$\lor$を定める。

  命題$\phi, \psi$について、命題$\phi \lor \psi$を以下で定める。
  \eq*{
    \phi \lor \psi \defiff \lnot \phi \rightarrow \psi
  }
}

\rem{命題結合子の優先順序}{
  結合の順序は以下で定めるものとする。
  \begin{enumerate}
    \item 否定$\lnot$
    \item 連言$\land$
    \item 選言$\lor$
    \item 含意$\rightarrow$
    \item 左含意$\leftarrow$
    \item 同値$\leftrightarrow$
  \end{enumerate}

  ただし$()$がある場合は、その中を先に結合する。
}

\cor{真理の導入}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \top
  }
}

\cor{否定の導入則}{
  命題$\phi, \chi$について、以下が成り立つ。
  \eq*{
    \chi \land \phi \vdash \bot \metaimply \chi \vdash \lnot \phi
  }
}

\cor{連言の単位}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \top \land \phi
  }
}

\cor{選言の導入則}{
  命題$\phi, \psi$について、以下の\num{2}つが成り立つ。
  \eqg*{
    \phi \vdash \phi \lor \psi \\*
    \psi \vdash \phi \lor \psi
  }
}

\cor{選言の除去則}{
  命題$\phi, \psi, \chi, \omega$について、以下が成り立つ。
  \eq*{
    \phi \land \psi \vdash \chi \metaand \phi \land \omega \vdash \chi \metaimply \phi \land \qty(\psi \lor \omega) \vdash \chi
  }
}

\cor{選言の冪等}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \phi \lor \phi
  }
}

\cor{選言の結合}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eqg*{
    \qty(\phi \lor \psi) \lor \chi \vdash \phi \lor \qty(\psi \lor \chi) \\*
    \phi \lor \qty(\psi \lor \chi) \vdash \qty(\phi \lor \psi) \lor \chi
  }
}

\cor{選言の交換}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \phi \lor \psi \vdash \psi \lor \phi
  }
}

\cor{選言三段論法}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \qty(\phi \lor \psi) \land \lnot \phi \vdash \psi
  }
}

\cor{吸収律}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eqg*{
    \vdash \phi \lor \qty(\phi \land \psi) \leftrightarrow \phi \\*
    \vdash \phi \land \qty(\phi \lor \psi) \leftrightarrow \phi
  }
}

\cor{分配律}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eqg*{
    \vdash \phi \lor \qty(\psi \land \chi) \leftrightarrow \qty(\phi \lor \psi) \land \qty(\phi \lor \chi) \\*
    \vdash \phi \land \qty(\psi \lor \chi) \leftrightarrow \qty(\phi \land \psi) \lor \qty(\phi \land \chi)
  }
}

\cor{De Morganの法則}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eqg*{
    \vdash \lnot \phi \lor \lnot \psi \leftrightarrow \lnot \qty(\phi \land \psi) \\*
    \vdash \lnot \phi \land \lnot \psi \leftrightarrow \lnot \qty(\phi \lor \psi)
  }
}

\cor{排中律}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \vdash \phi \lor \lnot \phi
  }
}

\cor{Dummettの法則}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \vdash \qty(\phi \rightarrow \psi) \lor \qty(\psi \rightarrow \phi)
  }
}


\lsubsection{述語論理}

\rem{述語論理の推論体系}{
  ここでは推論を行う上で許される規則として、
  後述する\axiref{全称の導入則}、\axiref{全称の除去則}を加える。
}

\rem{個体}{
  個体とは、数学的考察の対象とするものである。

  個体の全体を議論領域と呼ぶ。
}

\rem{変項}{
  変項とは、個体を指し示す仮の名前である。

  この仮の名前は、あなたが思いつく限り好きなものをつけることができる。

  後述する量化で扱うように、指し示す個体が具体的に定まっている必要はない。
}

\rem{項}{
  項とは、個体を指し示すものである。

  簡単には、変項と関数（後述）の値を合わせた概念である。
}

\dfn{述語}{
  \num{0}個以上の項の列について、その全てが確定したときに命題となる言明を、述語と呼ぶ。

  特に、命題結合子や後述する量化子を持たない述語を、原始述語と呼ぶ。

  述語$\phi$と、項の列$x, \ldots$を用いて$\phi(x, \ldots)$で表す。

  \vskip\baselineskip

  述語が要求する項の数を、その述語のアリティと呼ぶ。
}

\dfn{全称量化子}{
  変項$x$と、命題$\phi$を考える。

  このとき、$\forall x \qty(\phi)$は命題である。

  \vskip\baselineskip

  このとき$x$は、具体的な学問対象に解釈されないものとする。
  このような$x$は、束縛変項と呼ばれる。

  束縛変項でない変項を自由変項と呼ぶ。
}

\axi{全称の導入則}{
  命題$\phi, \psi$について、以下を定める。
  \eq*{
    \text{$\phi$が自由変項$x$を出現させない} \metaand \phi \vdash \psi \metaimply \phi \vdash \forall x \qty(\psi)
  }
}

\dfn{項への項の代入}{
  項$u$の変項$x$への項$t$の代入とは、新たな項を得る以下で定められる構造帰納的な操作である。
  これを、$u[x \coloneqq t]$で表す。

  \begin{enumerate}
    \item 項が変項$u$であるとき、代入の結果は$u = x$ならば$t$、さもなくば$u$である。
    \item 項が関数$f$を用いて$f(u, \ldots)$の形をとるとき、$f(u[x \coloneqq t], \ldots)$である。
  \end{enumerate}
}

\dfn{命題への項の代入}{
  命題$\phi$の変項$x$への項$t$の代入とは、新たな命題を得る以下で定められる構造帰納的な操作である。
  これを、$\phi[x \coloneqq t]$で表す。

  \begin{enumerate}
    \item 命題が$\bot$であるとき、代入の結果は$\bot$である。
    \item 命題が$\phi \land \psi$の形をとるとき、$\phi[x \coloneqq t] \land \psi[x \coloneqq t]$である。
    \item 命題が$\phi \rightarrow \psi$の形をとるとき、$\phi[x \coloneqq t] \rightarrow \psi[x \coloneqq t]$である。
    \item 命題が原始述語$P()$を用いて$P(u, \ldots)$の形をとるとき、$P\qty(u[x \coloneqq t], \ldots)$である。
    \item 命題が$\forall y \qty(\phi)$の形をとるとき、
    \begin{itemize}
      \item $x = y$であるとき、$\forall y \qty(\phi)$である。
      \item 項$t$が$y$を自由変項として出現させないとき、$\forall y \qty(\phi[x \coloneqq t])$である。
      \item いずれでもないとき、$t, \phi$に出現しない$x$とは異なる新しい変項$z$を持ってきて、$\forall z \qty(\phi[y \coloneqq z][x \coloneqq t])$である。
    \end{itemize}
  \end{enumerate}

  命題$\phi$は$x$を定めることで命題となる述語$\phi(x)$とみなすことができる。
  このとき、$\phi[x \coloneqq t]$は、$\phi(t)$となる。
}

\axi{全称の除去則}{
  項$t$、命題$\phi$について、以下を定める。
  \eq*{
    \forall x \qty(\phi) \vdash \phi[x \coloneqq t]
  }
}


\lsubsection{述語論理におけるいくつかの重要な概念と定理}

\cor{量化と演繹}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \phi \vdash \psi \metaimply \forall x \qty(\phi) \vdash \forall x \qty(\psi)
  }
}

\cor{アルファ同値}{
  自由変項$y$を出現させない命題$\phi$について、以下が成り立つ。
  \eq*{
    \vdash \forall x \qty(\phi) \leftrightarrow \forall y \qty(\phi[x \coloneqq y])
  }
}

\dfn{存在量化子}{
  変項$x$と、命題$\phi$について、存在量化子$\exists$を以下で定める。
  \eq*{
    \exists x \qty(\phi) \defiff \lnot \forall x \qty(\lnot \phi)
  }
}

\cor{存在の導入則}{
  変項$x$と、命題$\phi$について、以下が成り立つ。
  \eq*{
    \phi[x \coloneqq t] \vdash \exists x \qty(\phi)
  }
}

\cor{存在の除去則}{
  命題$\phi, \psi, \chi$について、以下が成り立つ。
  \eq*{
    \text{$\phi$と$\chi$が自由変項$x$を出現させない} \metaand \phi \land \psi \vdash \chi \metaimply \phi \land \exists x \qty(\psi) \vdash \chi
  }
}

\cor{量化子と否定}{
  命題$\phi$について、以下が成り立つ。
  \eqg*{
    \vdash \forall x \qty(\lnot \phi) \leftrightarrow \lnot \exists x \qty(\phi) \\*
    \vdash \lnot \forall x \qty(\phi) \leftrightarrow \exists x \qty(\lnot \phi)
  }
}

\cor{全称量化子と連言}{
  変項$x$と、命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \vdash \forall x \qty(\phi \land \psi) \leftrightarrow \forall x \qty(\phi) \land \forall x \qty(\psi)
  }
}

\cor{存在量化子と選言}{
  変項$x$と、命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \vdash \exists x \qty(\phi \lor \psi) \leftrightarrow \exists x \qty(\phi) \lor \exists x \qty(\psi)
  }
}

\cor{存在量化子と連言}{
  変項$x$と、命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \exists x \qty(\phi \land \psi) \vdash \exists x \qty(\phi) \land \exists x \qty(\psi)
  }
}

\cor{全称量化子と選言}{
  命題$\phi, \psi$について、以下が成り立つ。
  \eq*{
    \forall x \qty(\phi) \lor \forall x \qty(\psi) \vdash \forall x \qty(\phi \lor \psi)
  }
}

\cor{酒場の法則}{
  命題$\phi$について、以下が成り立つ。
  \eq*{
    \vdash \exists x \qty(\phi \rightarrow \forall y \qty(\phi[x \coloneqq y]))
  }
}

\rem{類についての略記}{
  略記$\forall ,$と$\exists ,$を、命題$\phi$について、以下で定める。
  \eqa*{
    \forall x, y \qty(\phi) &\defiff \forall x \forall y \qty(\phi) \\*
    \exists x, y \qty(\phi) &\defiff \exists x \exists y \qty(\phi)
  }

  \num{3}個以上の変項についても同様に定める。
}


\lsubsection{等号}

\dfn{等号}{
  アリティ\num{2}の述語記号$=$を定める。

  $=\qty(s, t)$を、簡単のために$s = t$で表す。
}

\axi{等号の反射律}{
  項$t$について、以下を定める。
  \eq*{
    \vdash t = t
  }
}

\axi{等号の代入原理}{
  変項$x$、命題$\phi$、項$s, t$について、以下を定める。
  \eq*{
    \phi[x \coloneqq s] \land s = t \vdash \phi[x \coloneqq t]
  }
}

\cor{等号の対称律}{
  項$s, t$について、以下が成り立つ。
  \eq*{
    s = t \vdash t = s
  }
}

\cor{等号の推移律}{
  項$s, t, u$について、以下が成り立つ。
  \eq*{
    s = t \land t = u \vdash s = u
  }
}

\dfn{等号否定}{
  アリティ\num{2}の述語記号$\neq$を以下で定める。
  \eq*{
    \neq \qty(s, t) \defiff \lnot \qty(s = t)
  }

  $\neq \qty(s, t)$を、簡単のために$s \neq t$で表す。
}

\cor{等号否定の対称律}{
  項$s, t$について、以下が成り立つ。
  \eq*{
    s \neq t \vdash t \neq s
  }
}

\rem{項の定義記号}{
  定義記号$\coloneqq$を導入する。

  これは左辺にある項は、議論において全てその左辺と一致する項を右辺に置き換えて理解するという意味である。
}

\dfn{一意存在量化子}{
  量化子$\exists!$を定める。

  変項$x$と、命題$\phi$について、一意存在量化子$\exists!$を以下で定める。
  \eq*{
    \exists! x \qty(\phi) \defiff \exists x \qty(\phi) \land \forall y, z \qty(\phi[x \coloneqq y] \land \phi[x \coloneqq z] \rightarrow y = z)
  }
}


\lsubsection{類と関数}

\dfn{類}{
  アリティ\num{1}の述語$\bm{A}$を、類と呼ぶ。

  類であることを強調するために、$\bm{A}(x)$を$x \in \bm{A}$と書く。
}

\dfn{類の否定}{
  類$\bm{A}$と項$x$について、記号$\notin$を以下で定める。
  \eq*{
    x \notin \bm{A} \defiff \lnot \qty(x \in \bm{A})
  }
}

\rem{類についての略記}{
  略記$\forall \in, \exists \in$を、類$X$と命題$\phi$について、以下で定める。
  \eqa*{
    \forall x \in X \qty(\phi) &\defiff \forall x \qty(x \in X \rightarrow \phi) \\*
    \exists x \in X \qty(\phi) &\defiff \exists x \qty(x \in X \land \phi)
  }
}

\rem{存在の誤謬}{
  類$X$と命題$\phi$について、
  $\forall x \in X \qty(\phi)$から$\exists x \in X \qty(\phi)$を導くことはできない。

  これは存在の誤謬と呼ばれる初歩的な誤謬の一種である。
}

\dfn{関数グラフ}{
  述語$D$について、以下を満たす述語$P$を、$D$上の関数グラフと呼ぶ。
  \eq*{
    \forall x \ldots \qty(D(x, \ldots) \rightarrow \exists! y \qty(P(x, \ldots, y)))
  }

  このとき、以下の略記$F$を定めて、$D$上の関数と呼ぶ。
  \eq*{
    \forall x \ldots \qty(D(x, \ldots) \rightarrow P(x, \ldots, F(x, \ldots)))
  }
}

\dfn{全域関数グラフ}{
  述語$D$が恒に真となるとき、$D$上の関数グラフ$P$を、全域関数グラフと呼ぶ。

  このとき、関数$F$を、全域関数と呼ぶ。
}

\axi{Skolem関係の導入}{
  述語$D$について、以下を満たす述語$P$を考える。
  \eq*{
    \forall x \ldots \qty(D(x, \ldots) \rightarrow \exists y \qty(P(x, \ldots, y)))
  }

  このとき、以下の\num{2}つを満たす新たな述語$Q$をとることができる。
  \eqg*{
    \forall x \ldots \qty(D(x, \ldots) \rightarrow \forall y \qty(Q(x, \ldots, y) \rightarrow P(x, \ldots, y))) \\*
    \forall x \ldots \qty(D(x, \ldots) \rightarrow \exists! y \qty(Q(x, \ldots, y)))
  }
}

\rem{Skolem関係の導入}{
  上の公理の導入により、通常の一階述語論理とは異なる論理体系となる。
}
