\lsection{圏論}

\lsubsection{射の公理}

\rem{対象と恒等射の同一視}{
  本ノートでは対象を恒等射と同一視する。
  一般的でないことに注意されたい。
}

\dfn{射}{
  圏論では、
  \num{2}つの全域関数（\dfnref{始域}、\dfnref{終域}）、
  \num{1}つの述語（\dfnref{合成可能性}）と、
  \num{6}つの公理（
    \axiref{始域の冪等性}、\axiref{終域の冪等性}、\axiref{合成}、\axiref{合成射の始域と終域}、\axiref{合成射の結合}、\axiref{恒等射}
  ）を満たすものを議論領域とする。

  圏論における項を射と呼ぶ。
}

\dfn{始域}{
  アリティ\num{1}の全域関数$\dom()$を考える。

  射$f$について、$\dom(f)$を$f$の始域と呼ぶ。
}

\axi{始域の冪等性}{
  \eq*{
    \forall f \qty(\dom(f) = \dom(\dom(f)) \land \dom(f) = \cod(\dom(f)))
  }
}

\dfn{終域}{
  アリティ\num{1}の全域関数$\cod()$を考える。

  射$f$について、$\cod(f)$を$f$の終域と呼ぶ。
}

\axi{終域の冪等性}{
  \eq*{
    \forall f \qty(\cod(f) = \dom(\cod(f)) \land \cod(f) = \cod(\cod(f)))
  }
}

\dfn{合成可能性}{
  アリティ\num{3}の述語$\Comp()$を考える。
}

\axi{合成}{
  \eq*{
    \forall f, g \qty(\cod(f) = \dom(g) \rightarrow \exists! h \qty(\Comp(f, g, h)))
  }
}

\dfn{合成}{
  $\Comp(f, g, h)$は、アリティ\num{2}の述語$\cod(f) = \dom(g)$上の関数グラフである。

  $h$を関数$g \circ f$で表す。
}

\axi{合成射の始域と終域}{
  \eq*{
    \forall f, g \qty(\cod(f) = \dom(g) \rightarrow \qty(\dom(g \circ f) = \dom(f) \land \cod(g \circ f) = \cod(g)))
  }
}

\axi{合成射の結合}{
  \eq*{
    \forall f, g, h \qty(\cod(f) = \dom(g) \land \cod(g) = \dom(h) \rightarrow h \circ \qty(g \circ f) = \qty(h \circ g) \circ f)
  }
}

\axi{恒等射}{
  \eq*{
    \forall f \qty(f = f \circ \dom(f) \land f = \cod(f) \circ f)
  }
}

\dfn{対象}{
  以下を満たす射$a$を、対象、または、恒等射と呼ぶ。
  \eq*{
    a = \dom(a)
  }
}

\cor{対象}{
  以下が成り立つ。
  \eq*{
    \forall a \qty(a = \dom(a) \leftrightarrow a = \cod(a))
  }
}


\lsubsection{圏と関手}

\dfn{圏}{
  類$\bm{C}$が圏であるとは、以下を満たすことである。
  \eqg*{
    \forall f \in \bm{C} \qty(\dom(f) \in \bm{C} \land \cod(f) \in \bm{C}) \\*
    \forall f, g \in \bm{C} \qty(\cod(f) = \dom(g) \rightarrow g \circ f \in \bm{C})
  }
}

\dfn{空圏}{
  以下で定める類$\bm{A}$を考える。
  \eq*{
    f \in \bm{A} \defiff \bot
  }

  このとき$\bm{A}$は、圏である。
  この$\bm{A}$を、空圏と呼ぶ。
}

\dfn{対象の類}{
  以下で定める類$\Obj_{\bm{C}}$を、圏$\bm{C}$の対象の類と呼ぶ。
  \eq*{
    f \in \Obj_{\bm{C}} \defiff f \in \bm{C} \land f = \dom(f)
  }
}

\dfn{離散圏}{
  圏$\bm{C}$が離散であるとは、以下を満たすことである。
  \eq*{
    \forall f \qty(f \in \bm{C} \leftrightarrow f \in \Obj_{\bm{C}})
  }

  このとき、$\bm{C}$を離散圏と呼ぶ。
}

\cor*{
  圏$\bm{C}$について、$\Obj_{\bm{C}}$は離散圏である。
}

\dfn{Hom類}{
  圏$\bm{C}$と、$\bm{C}$の対象$a, b$について、類$\Hom_{\bm{C}}(a, b)$を以下で定める。
  \eq*{
    f \in \Hom_{\bm{C}}(a, b) \defiff f \in \bm{C} \land \dom(f) = a \land \cod(f) = b
  }
}

\rem{射の域の明示}{
  $f \in \Hom_{\bm{C}}(a, b)$を、簡単のため、$\bm{C}$の射$f \colon a \to b$と表記する。
}

\dfn{End類}{
  圏$\bm{C}$と、$\bm{C}$の対象$a$について、類$\End_{\bm{C}}(a)$を以下で定める。
  \eq*{
    f \in \End_{\bm{C}}(a) \defiff f \in \Hom_{\bm{C}}(a, a)
  }
}

\dfn{関手}{
  圏$\bm{C}, \bm{D}$について、$\bm{C}$上の関数$F$が、以下の\num{3}つを満たすとき、$F$を$\bm{C}$から$\bm{D}$への関手と呼ぶ。
  \eqg*{
    \forall f \in \bm{C} \qty(F(f) \in \bm{D}) \\*
    \forall f \in \bm{C} \qty\Big(\dom(F(f)) = F(\dom(f)) \land \cod(F(f)) = F(\cod(f))) \\*
    \forall f, g \in \bm{C} \qty\Big(\cod(f) = \dom(g) \rightarrow F(g \circ f) = F(g) \circ F(f))
  }

  $\bm{C}$から$\bm{D}$への関手$F$を、$F \colon \bm{C} \to \bm{D}$で表す。
}

\cor*{
  圏$\bm{C}, \bm{D}$と関手$F \colon \bm{C} \to \bm{D}$について、以下が成り立つ。
  \eq*{
    \forall a \in \Obj_{\bm{C}} \qty(F(a) \in \Obj_{\bm{D}})
  }
}

\dfn{関手の合成}{
  圏$\bm{C}, \bm{D}, \bm{E}$と、関手$F \colon \bm{C} \to \bm{D}, G \colon \bm{D} \to \bm{E}$について、以下で定める$\bm{C}$上の関数$H$を考える。
  \eq*{
    H(f) \coloneqq G(F(f))
  }

  このとき$H$は、$\bm{C}$から$\bm{E}$への関手である。
  この$H$を、$F$と$G$の合成と呼び、$G \circ F$で表す。
}

\dfn{恒等関手}{
  圏$\bm{C}$について、以下で定める$\bm{C}$上の関数$F$を考える。
  \eq*{
    F(f) \coloneqq f
  }

  このとき関数$F$は、$\bm{C}$から$\bm{C}$への関手である。
  この$F$を、$\bm{C}$の恒等関手と呼び、$\id_{\bm{C}}$で表す。
}

\rem{射としての関手}{
  関手について、$=, \dom, \cod, \Comp$を以下で定めることにより、\subsecref{射の公理}を満たす。

  圏$\bm{C}, \bm{D}, \bm{E}, \bm{C^\prime}, \bm{D^\prime}$と、関手$F \colon \bm{C} \to \bm{D}, G \colon \bm{D} \to \bm{E}, H \colon \bm{C^\prime} \to \bm{D^\prime}$について、
  \begin{enumerate}
    \item $F = H \defiff \forall f \qty(f \in \bm{C} \leftrightarrow f \in \bm{C^\prime}) \land \forall f \qty(f \in \bm{D} \leftrightarrow f \in \bm{D^\prime}) \land \forall f \in \bm{C} \qty(F(f) = H(f))$
    \item $\dom(F) \coloneqq \id_{\bm{C}}$
    \item $\cod(F) \coloneqq \id_{\bm{D}}$
    \item $\circ$は\dfnref{関手の合成}の意味として、$\Comp(F, G, H) \defiff H = G \circ F$
  \end{enumerate}
}

\dfn{自明な関手}{
  圏$\bm{C}$について、以下で定める$\Obj_{\bm{C}}$上の関数$I$を考える。
  \eq*{
    I(a) \coloneqq a
  }

  このとき$I$は、$\Obj_{\bm{C}}$から$\bm{C}$への関手である。
  この$I$を、$\bm{C}$の自明な関手と呼ぶ。
}

\dfn{圏同型}{
  圏$\bm{C}, \bm{D}$を考える。
  関手$F \colon \bm{C} \to \bm{D}$について、関手$G \colon \bm{D} \to \bm{C}$が存在して、以下が成り立つとき、$F$は圏同型であるという。
  \eq*{
    G \circ F = \id_{\bm{C}} \land F \circ G = \id_{\bm{D}}
  }
}

\dfn{忠実}{
  圏$\bm{C}, \bm{D}$について、関手$F \colon \bm{C} \to \bm{D}$が以下を満たすとき、$F$は忠実であるという。
  \eq*{
    \forall a, b \in \Obj_{\bm{C}} \forall f, g \in \Hom_{\bm{C}}\qty(a, b) \qty(F(f) = F(g) \rightarrow f = g)
  }
}

\dfn{充満}{
  圏$\bm{C}, \bm{D}$について、関手$F \colon \bm{C} \to \bm{D}$が以下を満たすとき、$F$は充満であるという。
  \eq*{
    \forall a, b \in \Obj_{\bm{C}} \forall h \in \Hom_{\bm{D}}\qty(F(a), F(b)) \exists f \in \Hom_{\bm{C}}\qty(a, b) \qty(F(f) = h)
  }
}

\dfn{充満忠実}{
  圏$\bm{C}, \bm{D}$について、関手$F \colon \bm{C} \to \bm{D}$が充満であり忠実であるとき、$F$は充満忠実であるという。
}


\lsubsection{自然変換と関手圏}

\dfn{自然変換}{
  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime \colon \bm{C} \to \bm{D}$を考える。

  $\Obj_{\bm{C}}$上の関数$\eta$が、以下の\num{2}つを満たすとき、$\eta$を$F$から$F^\prime$への自然変換と呼ぶ。
  \eqg*{
    \forall a \in \Obj_{\bm{C}} \qty\Big(\eta(a) \in \Hom_{\bm{D}}(F(a), F^\prime(a))) \\*
    \forall f \in \bm{C} \qty\Big(F^\prime(f) \circ \eta(\dom(f)) = \eta(\cod(f)) \circ F(f))
  }

  $F$から$F^\prime$への自然変換$\eta$を、$\eta \colon F \to F^\prime$で表す。
}

\dfn{恒等な自然変換}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{C} \to \bm{D}$について、
  以下で定める$\Obj_{\bm{C}}$上の関数$\eta$を考える。
  \eq*{
    \eta(a) \coloneqq F(a)
  }

  このとき$\eta$は、$F$から$F$への自然変換である。
  この$\eta$を、$F$の恒等な自然変換と呼び、$\Id_F$で表す。
}

\dfn{自然変換の垂直合成}{
  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime, F^{\prime \prime} \colon \bm{C} \to \bm{D}$を考える。

  自然変換$\eta \colon F \to F^\prime, \theta \colon F^\prime \to F^{\prime \prime}$について、
  以下で定める$\Obj_{\bm{C}}$上の関数$\epsilon$を考える。
  \eq*{
    \qty(\theta \circ \eta)(a) \coloneqq \theta(a) \circ \eta(a)
  }

  このとき$\epsilon$は、$F$から$F^{\prime \prime}$への自然変換である。
  この$\epsilon$を、$\eta$と$\theta$の垂直合成と呼び、$\theta \circ \eta$で表す。
}

  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime \colon \bm{C} \to \bm{D}$を考える。

  自然変換$\eta \colon F \to F^\prime$について、以下が成り立つ。
  \eq*{
    \forall a \in \Obj_{\bm{C}} \qty\Big(\eta(a) = \qty(\eta \circ \Id_F)(a) \land \eta(a) = \qty(\Id_{F^\prime} \circ \eta)(a))
  }
}

\cor*{
  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime, F^{\prime \prime}, F^{\prime \prime \prime} \colon \bm{C} \to \bm{D}$を考える。

  自然変換$\eta \colon F \to F^\prime, \theta \colon F^\prime \to F^{\prime \prime}, \epsilon \colon F^{\prime \prime} \to F^{\prime \prime \prime}$について、
  以下が成り立つ。
  \eq*{
    \forall a \in \Obj_{\bm{C}} \qty\Big(\qty(\epsilon \circ \qty(\theta \circ \eta))(a) = \qty(\qty(\epsilon \circ \theta) \circ \eta)(a))
  }
}

\rem{射としての自然変換}{
  自然変換について、$=, \dom, \cod, \Comp$を以下で定めることにより、\subsecref{射の公理}を満たす。

  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime, F^{\prime \prime}, G, G^\prime \colon \bm{C} \to \bm{D}$、自然変換$\eta \colon F \to F^\prime, \theta \colon F^\prime \to F^{\prime \prime}, \epsilon \colon G \to G^\prime$について、
  \begin{enumerate}
    \item $\eta = \epsilon \defiff F = G \land F^\prime = G^\prime \land \forall a \in \Obj_{\bm{C}} \qty(\eta(a) = \epsilon(a))$
    \item $\dom(\eta) \coloneqq \Id_F$
    \item $\cod(\eta) \coloneqq \Id_{F^\prime}$
    \item $\circ$は\dfnref{自然変換の垂直合成}の意味として、$\Comp(\eta, \theta, \epsilon) \defiff \epsilon = \theta \circ \eta$
  \end{enumerate}
}

\dfn{関手圏}{
  圏$\bm{C}, \bm{D}$について、以下で定める類$\bm{A}$を考える。
  \eq*{
    \eta \in \bm{A} \defiff \text{自然変換$\eta \colon F \to G$、ただし関手$F, G \colon \bm{C} \to \bm{D}$であるもの}
  }

  このとき$\bm{A}$は、圏である。
  この$\bm{A}$を、$\bm{C}$から$\bm{D}$への関手圏と呼び、$\Func(\bm{C}, \bm{D})$で表す。
}

    \qty(\theta \ast \eta)(a) \coloneqq \theta(F^\prime(a)) \circ G(\eta(a))
  }

  このとき$\epsilon$は、$G \circ F$から$G^\prime \circ F^\prime$への自然変換である。
  この$\epsilon$を、$\eta$と$\theta$の水平合成と呼び、$\theta \ast \eta$で表す。
}

  自然変換$\eta \colon F \to F^\prime, \theta \colon G \to G^\prime$について、以下が成り立つ。
  \eq*{
    \forall a \in \Obj_{\bm{C}} \qty\Big(\qty(\theta \ast \Id_F)(a) = \theta(F(a)) \land \qty(\Id_G \ast \eta)(a) = G(\eta(a)))
  }
}

\lem*{
  圏$\bm{C}, \bm{D}, \bm{E}$と、関手$F, F^\prime \colon \bm{C} \to \bm{D}$、関手$G, G^\prime \colon \bm{D} \to \bm{E}$を考える。

  自然変換$\eta \colon F \to F^\prime, \theta \colon G \to G^\prime$について、以下が成り立つ。
  \eq*{
    \forall a \in \Obj_{\bm{C}} \qty\Big(\theta(F^\prime(a)) \circ G(\eta(a)) = G^\prime(\eta(a)) \circ \theta(F(a)))
  }
}{
  $\theta$は$G$から$G^\prime$への自然変換であるので、明らか。
}

\thm{相互交換法則}{
  圏$\bm{C}, \bm{D}, \bm{E}$と、関手$F, F^\prime, F^{\prime \prime} \colon \bm{C} \to \bm{D}$、関手$G, G^\prime, G^{\prime \prime} \colon \bm{D} \to \bm{E}$を考える。

  自然変換$\eta \colon F \to F^\prime, \theta \colon F^\prime \to F^{\prime \prime}, \epsilon \colon G \to G^\prime, \alpha \colon G^\prime \to G^{\prime \prime}$について、以下が成り立つ。
  \eq*{
    \qty(\alpha \circ \epsilon) \ast \qty(\theta \circ \eta) = \qty(\alpha \ast \theta) \circ \qty(\epsilon \ast \eta)
  }
}{
  $\bm{C}$の任意の対象$a$について、以下が成り立つ。
  \eqa*{
    \qty(\qty(\alpha \circ \epsilon) \ast \qty(\theta \circ \eta))(a)
    &= \qty(\alpha \circ \epsilon)\qty(F^{\prime \prime}(a)) \circ G \qty(\qty(\theta \circ \eta)(a)) \\*
    &= \alpha\qty(F^{\prime \prime}(a)) \circ \epsilon\qty(F^{\prime \prime}(a)) \circ G\qty(\theta(a)) \circ G \qty(\eta(a)) \\*
    &= \alpha\qty(F^{\prime \prime}(a)) \circ G^\prime\qty(\theta(a)) \circ \epsilon\qty(F^\prime(a)) \circ G \qty(\eta(a)) \\*
    &= \qty(\alpha \ast \theta)(a) \circ \qty(\epsilon \ast \eta)(a) \\*
    &= \qty(\qty(\alpha \ast \theta) \circ \qty(\epsilon \ast \eta))(a)
  }
}

\lsubsection{双対}

\dfn{反対射}{
  射$f$について、以下を満たす新たなもの$f^\opp$を考える。
  \eqg*{
    f = \dom(f) \rightarrow f^\opp = f \\*
    \qty(f^\opp)^\opp = f
  }

  $f^\opp$を$f$の反対射と呼ぶ。
}

\rem{射としての反対射}{
  反対射について、$=, \dom, \cod, \Comp$を以下で定めることにより、\subsecref{射の公理}を満たす。
  射$f, g, h$について、
  \begin{enumerate}
    \item $f^\opp = g^\opp \defiff f = g$
    \item $\dom(f^\opp) \coloneqq \cod(f)$
    \item $\cod(f^\opp) \coloneqq \dom(f)$
    \item $\Comp(f^\opp, g^\opp, h^\opp) \defiff h = f \circ g$
  \end{enumerate}
}

\dfn{反対圏}{
  圏$\bm{C}$について、以下で定める類$\bm{C}^\opp$を考える。
  \eq*{
    f \in \bm{C}^\opp \defiff f^\opp \in \bm{C}
  }

  このとき$\bm{C}^\opp$は、圏である。
  この$\bm{C}^\opp$を、$\bm{C}$の反対圏と呼ぶ。
}

\cor*{
  離散圏$\bm{J}$について、以下が成り立つ。
  \eq*{
    a \in \bm{J} \leftrightarrow a \in \bm{J}^\opp
  }
}

\dfn{反対関手}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{C} \to \bm{D}$について、
  以下で定める$\bm{C}^\opp$上の関数$F^\opp$を考える。
  \eq*{
    F^\opp(f) \coloneqq \qty(F(f^\opp))^\opp
  }

  このとき$F^\opp$は、$\bm{C}^\opp$から$\bm{D}^\opp$への関手である。
  この$F^\opp$を、$F$の反対関手と呼ぶ。
}

\dfn{反対自然変換}{
  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime \colon \bm{C} \to \bm{D}$を考える。

  自然変換$\eta \colon F \to F^\prime$について、以下で定める$\Obj_{\bm{C}}$上の関数$\eta^\opp$を考える。
  \eq*{
    \eta^\opp(a) \coloneqq \eta(a)^\opp
  }

  このとき$\eta^\opp$は$F^{\prime \opp}$から$F^\opp$への自然変換である。
  この$\eta^\opp$を、$\eta$の反対自然変換と呼ぶ。
}

\cor{関手圏の双対}{
  圏$\bm{C}, \bm{D}$について、以下で定める関手$F \colon \Func(\bm{C}, \bm{D})^\opp \to \Func(\bm{C}^\opp, \bm{D}^\opp)$は圏同型である。
  \eq*{
    F(\eta) \coloneqq \eta^\opp
  }
}

\rem{双対原理}{
  命題$\phi, \psi$と、双対化した命題$\phi^\opp, \psi^\opp$について、以下が成り立つ。
  \eq*{
    \phi \vdash \psi \metaimply \phi^\opp \vdash \psi^\opp
  }

  一般の命題について上が成り立つことの証明には、メタ論理が必要になるため、ここでは取り扱わない。

  証明部で「双対性より」という文脈に出会ったら、各自証明を補完されたい。
}


\lsubsection{簡約と逆}

\dfn{左簡約可能}{
  圏$\bm{C}$について、射$f \in \bm{C}$が以下を満たすとき、$f$は$\bm{C}$上で左簡約可能であるという。
  \eq*{
    \forall g, h \in \bm{C} \qty\Big(\cod(g) = \dom(f) \land \cod(h) = \dom(f) \land f \circ g = f \circ h \rightarrow g = h)
  }

  $\bm{C}$上で左簡約可能な射を、$\bm{C}$のモノ射と呼ぶ。
  特に圏$\bm{C}$が明らかな場合には、単にモノ射と呼ぶ。
}

\dfn{右簡約可能}{
  圏$\bm{C}$と射$f \in \bm{C}$について、$f^\opp$が$\bm{C}^\opp$で左簡約可能であるとき、
  $f$は$\bm{C}$上で右簡約可能であるという。

  $\bm{C}$上で右簡約可能な射を、$\bm{C}$のエピ射と呼ぶ。
  特に圏$\bm{C}$が明らかな場合には、単にエピ射と呼ぶ。

}

\dfn{左可逆}{
  圏$\bm{C}$について、射$f \in \bm{C}$が以下を満たすとき、$f$は$\bm{C}$上で左可逆であるという。
  \eq*{
    \exists g \in \bm{C} \qty(\cod(f) = \dom(g) \land g \circ f = \dom(f))
  }
}

\cor*{
  圏$\bm{C}$について、$\bm{C}$上で左可逆な射は、$\bm{C}$上で左簡約可能である。
}

\dfn{右可逆}{
  圏$\bm{C}$と射$f \in \bm{C}$について、$f^\opp$が$\bm{C}^\opp$で左可逆であるとき、
  $f$は$\bm{C}$上で右可逆であるという。
}

\cor*{
  圏$\bm{C}$について、$\bm{C}$上で右可逆な射は、$\bm{C}$上で右簡約可能である。
}

\cor*{
  圏$\bm{C}$と、射$f, g \in \bm{C}$について、$\cod(f) = \dom(g)$であるとする。
  このとき、以下が成り立つ。
  \begin{enumerate}
    \item $f, g$がともに$\bm{C}$上で左簡約可能ならば、$g \circ f$は$\bm{C}$上で左簡約可能である。
    \item $f, g$がともに$\bm{C}$上で右簡約可能ならば、$g \circ f$は$\bm{C}$上で右簡約可能である。
    \item $f, g$がともに$\bm{C}$上で左可逆ならば、$g \circ f$は$\bm{C}$上で左可逆である。
    \item $f, g$がともに$\bm{C}$上で右可逆ならば、$g \circ f$は$\bm{C}$上で右可逆である。
  \end{enumerate}
}

\cor*{
  圏$\bm{C}$と、射$f, g \in \bm{C}$について、$\cod(f) = \dom(g)$であるとする。
  このとき、以下が成り立つ。
  \begin{enumerate}
    \item $g \circ f$が$\bm{C}$上で左簡約可能ならば、$f$は$\bm{C}$上で左簡約可能である。
    \item $g \circ f$が$\bm{C}$上で右簡約可能ならば、$g$は$\bm{C}$上で右簡約可能である。
  \end{enumerate}
}

\dfn{逆射}{
  射$f$について、以下を満たす射$g$を$f$の逆射と呼ぶ。
  \eq*{
    \cod(f) = \dom(g) \land \cod(g) = \dom(f) \land g \circ f = \dom(f) \land f \circ g = \cod(f)
  }
}

\lem{逆射の一意性}{
  射$g, h$が$f$の逆射であるとき、$g = h$である。

  この補題から逆射は関数であり、$f$の逆射を$f^{-1}$で表す。
}{
  $g = g \circ \dom(g) = g \circ \cod(f) = g \circ f \circ h = \dom(f) \circ h = \cod(h) \circ h = h$より、成り立つ。
}

\cor*{
  射$f$が逆射を持つならば、以下が成り立つ。
  \eq*{
    \qty(f^{-1})^{-1} = f
  }
}

\lem{同型射}{
  圏$\bm{C}$と、射$f \in \bm{C}$について、以下の\num{2}つは同値である。
  \begin{enumerate}
    \item $f$は、$\bm{C}$に逆射を持つ。
    \item $f$は、$\bm{C}$上で、左可逆かつ右可逆である。
  \end{enumerate}
}{
  $1. \rightarrow 2.$は定義より明らか。

  \vskip\baselineskip

  $2. \rightarrow 1.$を示す。

  左可逆より、射$g$が存在して$g \circ f = \dom(f)$であり、右可逆より、射$h$が存在して$f \circ h = \cod(f)$である。

  $g = g \circ (f \circ h) = \qty(g \circ f) \circ h = h$であるので、$g = h$

  ゆえに、$g$は$f \circ g = \cod(f)$を満たす。
}

\dfn{同型射}{
  圏$\bm{C}$について、$\bm{C}$上で左可逆かつ右可逆な射を、$\bm{C}$上の同型射と呼ぶ。

  特に圏$\bm{C}$が明らかな場合には、単に同型射と呼ぶ。
}

\cor{恒等射は同型射}{
  圏$\bm{C}$について、恒等射$f \in \bm{C}$は、$\bm{C}$上の同型射である。
}

\cor*{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{C} \to \bm{D}$を考える。

  同型射$f \in \bm{C}$について、$F(f)$は$\bm{D}$上の同型射である。
}

\dfn{本質的全射}{
  圏$\bm{C}, \bm{D}$について、関手$F \colon \bm{C} \to \bm{D}$が以下を満たすとき、$F$は本質的全射であるという。
  \eq*{
    \forall d \in \Obj_{\bm{D}} \exists c \in \Obj_{\bm{C}} \qty(\text{同型射$F(c) \to d$が存在する})
  }
}

\lem{充満忠実ならば同型射を戻す}{
  圏$\bm{C}, \bm{D}$と、充満忠実な関手$F \colon \bm{C} \to \bm{D}$について、以下が成り立つ。
  \eq*{
    \forall f \in \bm{C} \qty(\text{$F(f)$が同型射} \rightarrow \text{$f$が同型射})
  }
}{
  $F$は充満忠実より、$\exists! g \in \Hom_{\bm{C}}\qty(\cod(f), \dom(f)) \qty(F(g) = F(f)^{-1})$

  $F$は忠実より、$F(g \circ f) = F(g) \circ F(f) = F(f)^{-1} \circ F(f) = \dom(F(f)) = F(\dom(f))$から、$g \circ f = \dom(f)$である。

  $F$は忠実より、$F(f \circ g) = F(f) \circ F(g) = F(f) \circ F(f)^{-1} = \cod(F(f)) = F(\cod(f))$から、$f \circ g = \cod(f)$である。
}

\dfn{自然同型}{
  圏$\bm{C}, \bm{D}$について、関手圏$\Func(\bm{C}, \bm{D})$上の同型射を、$\Func(\bm{C}, \bm{D})$上の自然同型と呼ぶ。

  特に関手圏$\Func(\bm{C}, \bm{D})$が明らかな場合には、単に自然同型と呼ぶ。
}

\cor*{
  圏$\bm{C}, \bm{D}$と、関手$F, F^\prime \colon \bm{C} \to \bm{D}$を考える。

  自然変換$\eta \colon F \to F^\prime$が自然同型であるとは、以下と同値である。
  \eq*{
    \forall a \in \Obj_{\bm{C}} \qty(\text{$\eta(a)$は同型射})
  }
}

\dfn{圏同値}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{C} \to \bm{D}$を考える。

  関手$G \colon \bm{D} \to \bm{C}$と、自然同型$\eta \colon \id_{\bm{C}} \to G \circ F, \theta \colon F \circ G \to \id_{\bm{D}}$が存在するとき、
  $F$は圏同値であるという。
}

\begin{center}
  \begin{minipage}{0.4\linewidth}
    $\bm{C}$ \\
    \begin{tikzcd}
      a \ar[rr, "f"] \ar[d, bend left, "\eta(a)"] & & b \ar[d, bend left, "\eta(b)"] \\
      \qty(G \circ F)(a) \ar[rr, "\qty(G \circ F)(f)"] \ar[u, bend left, "\eta(a)^{-1}"] & & \qty(G \circ F)(b) \ar[u, bend left, "\eta(b)^{-1}"] \\
    \end{tikzcd}
  \end{minipage}
  \begin{minipage}{0.4\linewidth}
    $\bm{D}$ \\
    \begin{tikzcd}
      c \ar[rr, "g"] \ar[d, bend left, "\theta(c)^{-1}"] & & d \ar[d, bend left, "\theta(d)^{-1}"] \\
      \qty(F \circ G)(c) \ar[rr, "\qty(F \circ G)(g)"] \ar[u, bend left, "\theta(c)"] & & \qty(F \circ G)(d) \ar[u, bend left, "\theta(d)"] \\
    \end{tikzcd}
  \end{minipage}
\end{center}

\thm{圏同値}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{C} \to \bm{D}$について、以下の\num{2}つは同値である。
  \begin{enumerate}
    \item $F$は圏同値
    \item $F$は充満忠実であり本質的全射である。
  \end{enumerate}
}{
  $1. \rightarrow 2.$を示す。

  定義より、関手$G \colon \bm{D} \to \bm{C}$と、自然同型$\eta \colon \id_{\bm{C}} \to G \circ F, \theta \colon F \circ G \to \id_{\bm{D}}$が存在する。

  \vskip\baselineskip

  対象$a, b \in \Obj_{\bm{C}}$と$\bm{C}$の射$f, g \colon a \to b$について、$F(f) = F(g)$とするとき、以下が成り立つ。
  \eqa*{
    f
    &= \eta(b)^{-1} \circ \eta(b) \circ f \\*
    &= \eta(b)^{-1} \circ \qty(G \circ F)(f) \circ \eta(a) \\*
    &= \eta(b)^{-1} \circ \qty(G \circ F)(g) \circ \eta(a) \\*
    &= \eta(b)^{-1} \circ \eta(b) \circ g \\*
    &= g
  }

  ゆえに忠実である。

  同様にして、$G$も忠実である。

  \vskip\baselineskip

  対象$a, b \in \Obj_{\bm{C}}$と$\bm{D}$の射$h \colon F(a) \to F(b)$について、以下が成り立つ。
  \eqa*{
    \qty(G \circ F)(\eta^{-1}(b) \circ G(h) \circ \eta(a))
    &= \eta(b) \circ \qty(\eta^{-1}(b) \circ G(h) \circ \eta(a)) \circ \eta(a)^{-1} \\*
    &= G(h)
  }

  $G$は忠実より、$F(\eta^{-1}(b) \circ G(h) \circ \eta(a)) = h$である。

  \vskip\baselineskip

  対象$d \in \Obj_{\bm{D}}$について、対象$G(d) \in \bm{C}$が存在して、同型射$\theta(d) \colon F(G(d)) \to d$が存在する。

  ゆえに、本質的全射である。

  \vskip\baselineskip

  $2. \rightarrow 1.$を示す。

  本質的全射であるので、以下を満たす\axiref{Skolem関係の導入}より$\Obj_{\bm{D}}$上の関数$G^\prime, \theta$を与えることができる。
  \eq*{
    \forall d \in \Obj_{\bm{D}} \qty(G^\prime(d) \in \Obj_{\bm{C}} \land \text{$\theta(d) \colon F(G^\prime(d)) \to d$は同型射})
  }

  射$f \in \bm{D}$について、$F$は充満忠実であるので、以下が成り立つ。
  \eq*{
    \exists! g \in \Hom_{\bm{C}}\qty(G^\prime(\dom(f)), G^\prime(\cod(f))) \qty(F(g) = \theta(\cod(f))^{-1} \circ f \circ \theta(\dom(f)))
  }

  上の命題が与える$\bm{D}$上の関数$G$を考える。

  $G$の定義より、$\forall f \in \bm{D} \qty(G(f) \in \bm{C})$である。

  任意の射$f \in \bm{D}$について、
  \eqa*{
    F(G(\dom(f)))
    &= \theta(\dom(f))^{-1} \circ \dom(f) \circ \theta(\dom(f)) \\*
    &= \theta(\dom(f))^{-1} \circ \theta(\dom(f)) \\*
    &= F(G^\prime(\dom(f))) \\*
    &= F(\dom(G(f)))
  }

  $F$は忠実より、$G(\dom(f)) = \dom(G(f))$である。

  同様にして、$G(\cod(f)) = \cod(G(f))$である。

  任意の射$f, g \in \bm{D}$について、$\dom(g) = \cod(f)$とすると、以下が成り立つ。
  \eqa*{
    F(G(g \circ f))
    &= \theta(\cod(g))^{-1} \circ g \circ f \circ \theta(\dom(f)) \\*
    &= \theta(\cod(g))^{-1} \circ g \circ \theta(\dom(g)) \circ \theta(\dom(g))^{-1} \circ f \circ \theta(\dom(f)) \\*
    &= \theta(\cod(g))^{-1} \circ g \circ \theta(\dom(g)) \circ \theta(\cod(f))^{-1} \circ f \circ \theta(\dom(f)) \\*
    &= \qty(F \circ G)(g) \circ \qty(F \circ G)(f) \\*
    &= F \qty(G(g) \circ G(f))
  }

  $F$は忠実より、$G(g \circ f) = G(g) \circ G(f)$である。

  ゆえに、$G$は$\bm{D}$から$\bm{C}$への関手である。

  $G$の定義より$\theta$は$F \circ G$から$\id_{\bm{D}}$への自然変換であり、$\theta$の定義より自然同型である。

  \vskip\baselineskip

  $F$は充満忠実であるので、以下を満たす。
  \eq*{
    \forall c \in \Obj_{\bm{C}} \exists! h \in \Hom_{\bm{C}}\qty(c, \qty(G \circ F)(c)) \qty(F(h) = \theta(F(c))^{-1})
  }

  上の命題が与える$\Obj_{\bm{C}}$上の関数$\eta$を考える。

  射$h \in \bm{C}$について、以下が成り立つ。
  \eqa*{
    F\qty(\qty(G \circ F)(h) \circ \eta(\dom(h)))
    &= \qty(F \circ G \circ F)(h) \circ F(\eta(\dom(h))) \\*
    &= \qty(F \circ G \circ F)(h) \circ \theta(F(\dom(h)))^{-1} \\*
    &= \theta(F(\cod(h)))^{-1} \circ F(h) \circ \theta(F(\dom(h))) \circ \theta(F(\dom(h)))^{-1} \\*
    &= \theta(F(\cod(h)))^{-1} \circ F(h) \\*
    &= F(\eta(\cod(h))) \circ F(h) \\*
    &= F(\eta(\cod(h)) \circ h)
  }

  $F$は忠実なので、$\qty(G \circ F)(h) \circ \eta(\dom(h)) = \eta(\cod(h)) \circ h$である。

  よって、$\eta$は$\id_{\bm{C}}$から$G \circ F$への自然変換である。

  $F$は充満忠実であるので、\lemref{充満忠実ならば同型射を戻す}より、$\eta$の定義より$\eta$は自然同型である。
}

\cor{圏同型ならば圏同値}{
  圏$\bm{C}, \bm{D}$と関手$F \colon \bm{C} \to \bm{D}$について、$F$が圏同型ならば$F$は圏同値である。
}


\lsubsection{普遍射}

\dfn{普遍射}{

  $F$から$a$への普遍射とは、対象$x \in \bm{D}$と$\bm{C}$の射$u \colon F(x) \to a$の組であって、以下を満たすものである。
  \eq*{
    \forall y \in \Obj_{\bm{D}} \forall f \in \Hom_{\bm{C}}\qty(F(y), a) \exists! g \in \Hom_{\bm{D}}(y, x) \qty(f = u \circ F(g))
  }
}

\cor{普遍射に同型ならば普遍射}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{D} \to \bm{C}$、対象$a \in \Obj_{\bm{C}}$を考える。

  $F$から$a$への普遍射$x, u$と、同型射$h \in \bm{D}$が存在して、$\cod(h) = x$であるとき、
  $\dom(h), u \circ F(h)$は、$F$から$a$への普遍射である。
}

\thm{普遍射の同型一意性}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{D} \to \bm{C}$、対象$a \in \Obj_{\bm{C}}$を考える。

  $x, u$と$x^\prime, u^\prime$が、ともに$F$から$a$への普遍射であるとき、以下を満たす。
  \eq*{
    \exists! h \in \Hom_{\bm{D}}(x^\prime, x) \qty(u^\prime = u \circ F(h))
  }

  さらに上で与える$h$は同型射である。
}{
  $x, u$は普遍射であるので、$x^\prime, u^\prime$について、
  $\bm{D}$の射$g \colon x^\prime \to x$が一意に存在して、$u^\prime = u \circ F(g)$を満たす。

  同様に$x^\prime, u^\prime$は普遍射であるので、$x, u$について、
  $\bm{D}$の射$g^\prime \colon x \to x^\prime$が一意に存在して、$u = u^\prime \circ F(g^\prime)$を満たす。

  ゆえに、$u = u^\prime \circ F(g^\prime) = u \circ F(g) \circ F(g^\prime) = u \circ F(g \circ g^\prime)$が成り立つ。

  $x, u$は普遍射であるので、$x, u$について、一意性より、$x = g \circ g^\prime$である。

  同様に、$x^\prime = g^\prime \circ g$である。

  したがって、$g^\prime$は$g$の逆射である。
}

  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{D} \to \bm{C}$、対象$a \in \Obj_{\bm{C}}$を考える。

  $F^\opp$から$a$への普遍射$x, u$について、$x, u^\opp$を$a$から$F$への余普遍射と呼ぶ。
}

\dfn{核対}{

\cor*{

\lem{核対と左簡約可能性}{
  圏$\bm{C}$と、射$f \in \bm{C}$について、$f$の核対$R, \qty(\pi_0, \pi_1)$が存在するとする。

  このとき、以下の\num{3}つは同値である。
  \begin{enumerate}
    \item $f$は左簡約可能
    \item $\pi_0 = \pi_1$
    \item $\pi_0, \pi_1$はともに同型射
  \end{enumerate}
}{
  $1. \rightarrow 2.$を示す。

  核対より$f \circ \pi_0 = f \circ \pi_1$であり、左簡約可能より$\pi_0 = \pi_1$である。

  \vskip\baselineskip

  $2. \rightarrow 3.$を示す。

  $f \circ \dom(f) = f \circ \dom(f)$より、普遍性から射$u \colon \dom(f) \to R$が存在して、$\pi_0 \circ u = \dom(f) = \pi_1 \circ u$である。

  $\pi_0 = \pi_1$より、$\pi_1 = \pi_0 = \pi_1 \circ u \circ \pi_0 = \pi_0 \circ u \circ \pi_0$である。

  $f \circ \pi_0 = f \circ \pi_0$より、普遍性から$R = u \circ \pi_0$である。

  したがって、$\pi_0$は同型射である。

  \vskip\baselineskip

  $3. \rightarrow 2.$を示す。

  $f \circ \pi_0 = f \circ \pi_0$より、一意な射$e \colon R \to R$が存在して、$\pi_0 = \pi_0 \circ e = \pi_1 \circ e$である。

  $\pi_0$は同型射より、$e = R$である。ゆえに、$\pi_0 = \pi_1$である。

  \vskip\baselineskip

  $2. \to 1.$を示す。

  射$g, h \in \bm{C}$について、$f \circ g = f \circ h$とする。

  普遍性より、一意な射$q \colon \dom(g) \to R$が存在して、$g = \pi_0 \circ q \land h = \pi_1 \circ q$である。

  $\pi_0 = \pi_1$より、$g = h$である。
}

\lem{等化子から得るモノ射}{
  $f$と$g$の等化子$\Eq(f, g), \pi$が存在するとする。

  このとき、$\pi$は左簡約可能である。
}{
  $\pi \circ h_1 = \pi \circ h_2$なる射$h_1, h_2 \colon y \to \Eq(f, g)$と、射$w \coloneqq \pi \circ h_1$を考える。

  このとき、$f \circ w = f \circ \pi \circ h_1 = g \circ \pi \circ h_1 = g \circ w$より、等化子の普遍性から$w = \pi \circ h$なる$h$が一意に存在する。

  ゆえに、$h_1 = h_2$である。

  したがって、$\pi$は左簡約可能である。
}

\dfn{対象としての射}{

\rem{対象としての射}{
  対象としての射について、$=, \dom, \cod, \Comp$を以下で定めることにより、\subsecref{射の公理}を満たす。
  射$f, g, h$について、
  \begin{enumerate}
    \item $\hat{f} = \hat{g} \defiff f = g$
    \item $\dom(\hat{f}) \coloneqq \hat{f}$
    \item $\cod(\hat{f}) \coloneqq \hat{f}$
    \item $\Comp(\hat{f}, \hat{g}, \hat{h}) \defiff \hat{h} = \hat{f} \land \hat{h} = \hat{g}$
  \end{enumerate}
}

\thm{極限の存在定理}{
  以下を満たす圏$\bm{J}, \bm{C}$を考える。
  \begin{enumerate}
    \item $\Obj_{\bm{J}}$から$\bm{C}$への任意の関手は、積を持つ。
    \item $\bm{J}$の射の上の離散圏$\hat{\bm{J}}$について、$\hat{\bm{J}}$から$\bm{C}$への任意の関手は、積を持つ。
    \item $\bm{C}$の任意の射$f, g$について$\dom(f) = \dom(g) \land \cod(f) = \cod(g)$ならば、等化子を持つ。
  \end{enumerate}

  このとき、関手$X \colon \bm{J} \to \bm{C}$は極限を持つ。
}{
  以下で定める$\hat{\bm{J}}$上の関数$S, T$を考える。
  \eqa*{
    S(\hat{i}) &\coloneqq \dom(i) \\*
    T(\hat{i}) &\coloneqq \cod(i)
  }

  $S, T$はともに$\hat{\bm{J}}$から$\Obj_{\bm{J}}$への関手である。

  以下で定める$\hat{\bm{J}}$上の関数$\alpha$を考える。
  \eq*{
    \alpha(\hat{i}) \coloneqq i
  }

  $\alpha$は、$I \circ S$から$I \circ T$への自然変換である。

  \vskip\baselineskip

  自明な関手$I \colon \Obj_{\bm{J}} \to \bm{J}$、
  対角関手$\Delta \colon \bm{C} \to \Func(\bm{J}, \bm{C})$、
  対角関手$\hat{\Delta} \colon \bm{C} \to \Func(\hat{\bm{J}}, \bm{C})$を考える。

  仮定より、$X \circ I$の積$x, \pi$と、$X \circ I \circ T$の積$\hat{x}, \hat{\pi}$が存在する。

  積の定義より、一意な射$f \colon x \to \hat{x}$が存在して、$\pi \ast \Id_T = \hat{\pi} \circ \hat{\Delta}(f)$である。

  積の定義より、一意な射$g \colon x \to \hat{x}$が存在して、$\qty(\Id_X \ast \alpha) \circ \qty(\pi \ast \Id_S) = \hat{\pi} \circ \hat{\Delta}(g)$である。

  仮定より、$f, g$の等化子$\Eq(f, g), u$が存在する。

  \vskip\baselineskip

  $\Obj_{\bm{J}}$上の関数$\lambda$を以下で定める。
  \eq*{
    \lambda(j) \coloneqq \pi(j) \circ u
  }

  射$i \in \bm{J}$について、等化子と積の定義より以下が成り立つ。
  \eqa*{
    X(i) \circ \lambda(\dom(i))
    &= X(i) \circ \pi(\dom(i)) \circ u \\*
    &= \qty(\qty(\Id_X \ast \alpha) \circ \qty(\pi \ast \Id_S))\qty(\hat{i}) \circ u \\*
    &= \hat{\pi}\qty(\hat{i}) \circ g \circ u \\*
    &= \hat{\pi}\qty(\hat{i}) \circ f \circ u \\*
    &= \qty(\pi \ast \Id_T)\qty(\hat{i}) \circ u \\*
    &= \pi\qty(\cod(i)) \circ u \\*
    &= \lambda\qty(\cod(i))
  }

  よって、$\lambda$は$\Delta\qty(\Eq(f, g))$から$X$への自然変換である。

  \vskip\baselineskip

  $\Eq(f, g), \lambda$が$X$の極限であることを示す。

  対象$y \in \bm{C}$と、自然変換$p \colon \Delta(y) \to X$を考える。

  $x, \pi$は$X \circ I$の積であるので、一意な射$w \colon y \to x$が存在して、以下が成り立つ。
  \eq*{
    \forall a \in \Obj_{\bm{J}} \qty(p(a) = \pi(a) \circ w)
  }

  任意の射$i \in \bm{J}$について、$p$の自然性より以下が成り立つ。
  \eqa*{
    \hat{\pi}\qty(\hat{i}) \circ f \circ w
    &= \qty(\pi \ast \Id_T)\qty(\hat{i}) \circ w \\*
    &= \pi\qty(\cod(i)) \circ w \\*
    &= p\qty(\cod(i)) \\*
    &= X(i) \circ p\qty(\dom(i)) \\*
    &= X(i) \circ \pi\qty(\dom(i)) \circ w \\*
    &= \qty(\qty(\Id_X \ast \alpha) \circ \qty(\pi \ast \Id_S))\qty(\hat{i}) \circ w \\*
    &= \hat{\pi}\qty(\hat{i}) \circ g \circ w
  }

  $\hat{x}, \hat{\pi}$は$X \circ I \circ T$の積であるので、普遍性より$f \circ w = g \circ w$である。

  $\Eq(f, g), u$は等化子より、一意な射$h \colon y \to \Eq(f, g)$が存在して、$w = u \circ h$を満たす。

  このとき、任意の対象$j \in \Obj_{\bm{J}}$について、以下が成り立つ。
  \eqa*{
    \qty(\lambda \circ \Delta(h))(j)
    &= \lambda(j) \circ h \\*
    &= \pi(j) \circ u \circ h \\*
    &= \pi(j) \circ w \\*
    &= p(j)
  }

  ゆえに、$p = \lambda \circ \Delta(h)$が成り立つ。

  \vskip\baselineskip

  一意性を示す。

  射$h^\prime \colon y \to \Eq(f, g)$が存在して、$p = \lambda \circ \Delta(h^\prime)$を満たすとする。

  このとき、任意の対象$j \in \Obj_{\bm{J}}$について以下が成り立つ。
  \eqa*{
    \pi(j) \circ u \circ h^\prime
    &= \lambda(j) \circ h^\prime \\*
    &= \qty(\lambda \circ \Delta(h^\prime))(j) \\*
    &= p(j) \\*
    &= \pi(j) \circ w \\*
    &= \pi(j) \circ u \circ h
  }

  積の普遍性より$u \circ h^\prime = u \circ h$であり、等化子の普遍性より$h^\prime = h$である。
}

\lsubsection{随伴}

\dfn{左随伴関手}{
  圏$\bm{C}, \bm{D}$について、以下を満たす関手$F \colon \bm{D} \to \bm{C}$を左随伴関手と呼ぶ。
  \eq*{
    \forall c \in \Obj_{\bm{C}} \qty(\text{$F$から$c$への普遍射が存在する})
  }
}

\dfn{右随伴関手}{
  圏$\bm{C}, \bm{D}$と関手$G \colon \bm{C} \to \bm{D}$について、$G^\opp$が左随伴関手であるとき、$G$を右随伴関手と呼ぶ。

\thm{左随伴関手}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{D} \to \bm{C}$について、以下の\num{4}つは同値である。
  \begin{enumerate}
    \item $F$は左随伴関手である。
    \item 関手$G \colon \bm{C} \to \bm{D}$と自然変換$\epsilon \colon F \circ G \to \id_{\bm{C}}$が存在して、以下が成り立つ。
    \begin{itemize}
      \item $\forall c \in \Obj_{\bm{C}} \qty(\text{$G(c), \epsilon(c)$は、$F$から$c$への普遍射である})$
    \end{itemize}
    \item 関手$G \colon \bm{C} \to \bm{D}$と自然変換$\epsilon \colon F \circ G \to \id_{\bm{C}}$が存在して、以下が成り立つ。
    \begin{itemize}
      \item $\forall c \in \Obj_{\bm{C}} \forall d \in \Obj_{\bm{D}} \forall f \in \Hom_{\bm{C}}\qty(F(d), c) \exists! g \in \Hom_{\bm{D}}\qty(d, G(c)) \qty(f = \epsilon(c) \circ F(g))$
    \end{itemize}
    \item 関手$G \colon \bm{C} \to \bm{D}$と自然変換$\epsilon \colon F \circ G \to \id_{\bm{C}}, \eta \colon \id_{\bm{D}} \to G \circ F$が存在して、以下が成り立つ。
    \begin{itemize}
      \item $\Id_F = \qty(\epsilon \ast \Id_F) \circ \qty(\Id_F \ast \eta)$
      \item $\Id_G = \qty(\Id_G \ast \epsilon) \circ \qty(\eta \ast \Id_G)$
    \end{itemize}
  \end{enumerate}
}{
  $1. \rightarrow 2.$を示す。

  \axiref{Skolem関係の導入}より以下を満たす$\Obj_{\bm{C}}$上の関数$G^\prime, \epsilon$を与えることができる。
  \eq*{
    \forall c \in \Obj_{\bm{C}} \qty(\text{$G^\prime(c), \epsilon(c)$は、$F$から$c$への普遍射である})
  }

  射$f \in \bm{C}$について、普遍性より$\bm{D}$の一意な射$g \colon G^\prime(\dom(f)) \to G^\prime(\cod(f))$が存在して、$f \circ \epsilon(\dom(f)) = \epsilon(\cod(f)) \circ F(g)$である。

  この命題が与える$\bm{C}$上の関数$G$を考える。

  $G$の定義より$\forall f \in \bm{C} \qty(G(f) \in \bm{D})$である。

  射$f \in \bm{C}$について、$G$の定義より$\epsilon(\dom(f)) = \dom(f) \circ \epsilon(\dom(f)) = \epsilon(\dom(f)) \circ F(G(\dom(f)))$である。

  $\epsilon(\dom(f)) = \epsilon(\dom(f)) \circ F(G^\prime(\dom(f)))$であるので、普遍性より$G(\dom(f)) = G^\prime(\dom(f)) = \dom(G(f))$である。

  同様に$G(\cod(f)) = \cod(G(f))$である。

  射$f, g \in \bm{C}$について、$\cod(f) = \dom(g)$とすると、以下が成り立つ。
  \eqa*{
    \epsilon(\cod(g)) \circ F(G(g) \circ G(f))
    &= \epsilon(\cod(g)) \circ F(G(g)) \circ F(G(f)) \\*
    &= g \circ \epsilon(\dom(g)) \circ F(G(f)) \\*
    &= g \circ \epsilon(\cod(f)) \circ F(G(f)) \\*
    &= g \circ f \circ \epsilon(\dom(f)) \\*
    &= \epsilon(\cod(g)) \circ F(G(g \circ f))
  }

  普遍性より、$G(g \circ f) = G(g) \circ G(f)$である。

  よって、$G$は$\bm{C}$から$\bm{D}$への関手である。

  $G$の定義より$\epsilon$は$F \circ G$から$\id_{\bm{C}}$への自然変換である。

  \vskip\baselineskip

  $2. \rightarrow 1.$は、明らか。

  \vskip\baselineskip

  $2. \rightarrow 3.$は、普遍射の定義より明らか。

  \vskip\baselineskip

  $3. \rightarrow 4.$を示す。

  $3.$について、$c = F(d), f = F(d)$として、以下の命題を満たす。
  \eq*{
    \forall d \in \Obj_{\bm{D}} \exists! g \in \Hom_{\bm{D}}\qty(d, \qty(G \circ F)(d)) \qty(F(d) = \epsilon\qty(F(d)) \circ F(g))
  }

  この命題より定まる$\Obj_{\bm{D}}$上の関数を$\eta$とする。

  \vskip\baselineskip

  $\eta$が自然変換であることを示す。

  $\bm{D}$の射$h \colon d \to d^\prime$は以下を満たす。
  \eqa*{
    \epsilon\qty(F(d^\prime)) \circ F\qty(\qty(G \circ F)(h) \circ \eta(d))
    &= \epsilon \qty(F(d^\prime)) \circ \qty(F \circ G \circ F)(h) \circ F\qty(\eta(d)) \\*
    &= F(h) \circ \epsilon\qty(F(d)) \circ F\qty(\eta(d)) \\*
    &= F(h) \circ F(d) \\*
    &= F(h) \\*
    &= F(d^\prime) \circ F(h) \\*
    &= \epsilon\qty(F(d^\prime)) \circ F\qty(\eta(d^\prime)) \circ F(h) \\*
    &= \epsilon\qty(F(d^\prime)) \circ F\qty(\eta(d^\prime) \circ h)
  }

  一意性より、$\qty(G \circ F)(h) \circ \eta(d) = \eta(d^\prime) \circ h$である。

  ゆえに、$\eta$は自然変換である。

  $\eta$の定義より、第一式を満たす。

  第二式を示す。

  対象$c \in \Obj_{\bm{C}}$を考える。

  $\epsilon$は自然変換であるので、射$w \colon \qty(F \circ G)(c) \to c$について$\epsilon(c) \circ \qty(F \circ G)(w) = w \circ \epsilon\qty(\qty(F \circ G)(c))$である。

  $w = \epsilon(c)$として、$\epsilon(c) \circ \qty(F \circ G)(\epsilon(c)) = \epsilon(c) \circ \epsilon\qty(\qty(F \circ G)(c))$であるので、
  以下が成り立つ。
  \eqa*{
    \epsilon(c) \circ F\qty(G(c))
    &= \epsilon(c) \circ \epsilon\qty(F\qty(G(c))) \circ F\qty(\eta(G(c))) \\*
    &= \epsilon(c) \circ \qty(F \circ G)\qty(\epsilon(c)) \circ F\qty(\eta(G(c))) \\*
    &= \epsilon(c) \circ F \qty\big(\qty(\qty(\Id_G \ast \epsilon) \circ \qty(\eta \ast \Id_G))(c))
  }

  一意性より、$\qty(\qty(\Id_G \ast \epsilon) \circ \qty(\eta \ast \Id_G))(c) = \Id_G(c)$である。

  \vskip\baselineskip

  $4. \rightarrow 2.$を示す。

  対象$a \in \bm{C}$について、$G(a), \epsilon(a)$が$F$から$a$への普遍射となることを示す。

  対象$y \in \bm{D}$と、$\bm{C}$の射$f \colon F(y) \to a$と、$\bm{D}$の射$g \coloneqq G(f) \circ \eta(y)$について、以下が成り立つ。
  \eqa*{
    \epsilon(a) \circ F(g)
    &= \epsilon(a) \circ \qty(F \circ G)(f) \circ \qty(\Id_F \ast \eta)(y) \\*
    &= f \circ \qty(\epsilon \ast \Id_F)(y) \circ \qty(\Id_F \ast \eta)(y) \\*
    &= f \circ F(y) \\*
    &= f
  }

  次に、$\bm{D}$の射$g^\prime \colon y \to G(a)$が存在して、$f = \epsilon(a) \circ F(g^\prime)$であるとする。

  このとき、以下が成り立つ。
  \eqa*{
    g^\prime
    &= G(a) \circ g^\prime \\*
    &= \qty(\Id_G \ast \epsilon)(a) \circ \qty(\eta \ast \Id_G)(a) \circ g^\prime \\*
    &= \qty(\Id_G \ast \epsilon)(a) \circ \qty(G \circ F)(g^\prime) \circ \eta(y) \\*
    &= G \qty(\epsilon(a) \circ F(g^\prime)) \circ \eta(y) \\*
    &= G(f) \circ \eta(y) \\*
    &= g
  }

  ゆえに一意である。
}

\lem{随伴の同型一意性}{
  圏$\bm{C}, \bm{D}$と、関手$F \colon \bm{D} \to \bm{C}$を考える。

  関手$G, G^\prime \colon \bm{C} \to \bm{D}$、
  自然変換$\epsilon \colon F \circ G \to \id_{\bm{C}}, \epsilon^\prime \colon F \circ G^\prime \to \id_{\bm{C}}$が、
  以下を満たすとする。
  \eqg*{
    \forall c \in \Obj_{\bm{C}} \qty(\text{$G(c), \epsilon(c)$は、$F$から$c$への普遍射である}) \\*
    \forall c \in \Obj_{\bm{C}} \qty(\text{$G^\prime(c), \epsilon^\prime(c)$は、$F$から$c$への普遍射である})
  }

  このとき自然同型$\theta \colon G \to G^\prime$が存在して、以下が成り立つ。
  \eq*{
    \epsilon^\prime \circ \qty(\Id_F \ast \theta) = \epsilon
  }
}{
  \thmref{左随伴関手}より、自然変換$\eta \colon \id_{\bm{D}} \to G \circ F, \eta^\prime \colon \id_{\bm{D}} \to G^\prime \circ F$が存在して、以下を満たす。
  \eqa*{
    \Id_F &= \qty(\epsilon \ast \Id_F) \circ \qty(\Id_F \ast \eta) \\*
    \Id_G &= \qty(\Id_G \ast \epsilon) \circ \qty(\eta \ast \Id_G) \\*
    \Id_F &= \qty(\epsilon^\prime \ast \Id_F) \circ \qty(\Id_F \ast \eta^\prime) \\*
    \Id_{G^\prime} &= \qty(\Id_{G^\prime} \ast \epsilon^\prime) \circ \qty(\eta^\prime \ast \Id_{G^\prime})
  }

  自然変換$\theta \coloneqq \qty(\Id_{G^\prime} \ast \epsilon) \circ \qty(\eta^\prime \ast \Id_G)$を考える。

  対象$c \in \Obj_{\bm{C}}$について、以下を満たす。
  \eqa*{
    \epsilon^\prime(c) \circ \qty(\Id_F \ast \theta)(c)
    &= \epsilon^\prime(c) \circ F\qty\big(G^\prime(\epsilon(c)) \circ \eta^\prime(G(c))) \\*
    &= \epsilon^\prime(c) \circ \qty(F \circ G^\prime)(\epsilon(c)) \circ \qty(\Id_F \ast \eta^\prime)(G(c)) \\*
    &= \epsilon(c) \circ \epsilon^\prime\qty(\qty(F \circ G)(c)) \circ \qty(\Id_F \ast \eta^\prime)(G(c)) \\*
    &= \epsilon(c) \circ F(G(c)) \\*
    &= \epsilon(c)
  }

  よって、$\epsilon^\prime \circ \qty(\Id_F \ast \theta) = \epsilon$である。

  同様に、自然変換$\theta^\prime \coloneqq \qty(\Id_G \ast \epsilon^\prime) \circ \qty(\eta \ast \Id_{G^\prime})$について、
  $\epsilon \circ \qty(\Id_F \ast \theta^\prime) = \epsilon^\prime$が成り立つ。

  ゆえに、以下を満たす。
  \eqa*{
    \epsilon(c) \circ F(G(c)) &= \epsilon(c) = \epsilon(c) \circ F\qty(\theta^\prime(c) \circ \theta(c)) \\*
    \epsilon^\prime(c) \circ F(G^\prime(c)) &= \epsilon^\prime(c) = \epsilon^\prime(c) \circ F\qty(\theta(c) \circ \theta^\prime(c)) \\*
  }

  \thmref{左随伴関手}より、一意性から$\theta(c)$は逆射$\theta^\prime(c)$を持つ。
}

\lem{圏同値ならば随伴}{
  圏$\bm{C}, \bm{D}$と、圏同値$F \colon \bm{D} \to \bm{C}$について、$F$は左随伴関手である。
}{
  圏同値の定義より、
  関手$G \colon \bm{C} \to \bm{D}$と、自然同型$\epsilon \colon F \circ G \to \id_{\bm{C}}, \theta \colon \id_{\bm{D}} \to G \circ F$が存在する。

  対象$c \in \Obj_{\bm{C}}$について、$G(c), \epsilon(c)$が$F$から$c$への普遍射であることを示す。

  対象$d \in \Obj_{\bm{D}}$と、$\bm{C}$の射$f \colon F(d) \to c$を考える。

  \thmref{圏同値}より$F$は充満であるので、
  $\bm{D}$の射$g \colon d \to G(c)$が存在して、$F(g) = \epsilon(c)^{-1} \circ f$である。

  よって、$f = \epsilon(c) \circ \epsilon(c)^{-1} \circ f = \epsilon(c) \circ F(g)$である。

  $\bm{D}$の射$g^\prime \colon d \to G(c)$が存在して、$f = \epsilon(c) \circ F(g^\prime)$とする。

  $F(g) = \epsilon(c)^{-1} \circ f = F(g^\prime)$であり、
  \thmref{圏同値}より$F$は忠実であるので、$g = g^\prime$である。

  ゆえに、$G(c), \epsilon(c)$は$F$から$c$への普遍射である。
}

\thm{左随伴は余極限を保存する}{
  圏$\bm{C}, \bm{D}, \bm{J}$と、関手$X \colon \bm{J} \to \bm{D}$、左随伴関手$F \colon \bm{D} \to \bm{C}$を考える。

  $X$の余極限$x, u$について、$F(x), \Id_F \ast u$は$F \circ X$の余極限である。
}{
  対角関手$\Delta_{\bm{D}} \colon \bm{D} \to \Func(\bm{J}, \bm{D}), \Delta_{\bm{C}} \colon \bm{C} \to \Func(\bm{J}, \bm{C})$を考える。

  対象$c \in \bm{C}$と、自然変換$\theta \colon F \circ X \to \Delta_{\bm{C}}(c)$を考える。

  \vskip\baselineskip

  \thmref{左随伴関手}より、関手$G \colon \bm{C} \to \bm{D}$と自然変換$\epsilon \colon F \circ G \to \id_{\bm{C}}$が存在して、以下が成り立つ。
  \eq*{
    \forall c \in \Obj_{\bm{C}} \forall d \in \Obj_{\bm{D}} \forall f \in \Hom_{\bm{C}}\qty(F(d), c) \exists! g \in \Hom_{\bm{D}}\qty(d, G(c)) \qty(f = \epsilon(c) \circ F(g))
  }

  したがって、任意の対象$j \in \Obj_{\bm{J}}$について、射$g \in \Hom_{\bm{D}}\qty(X(j), G(c))$が一意に存在して、$\theta(j) = \epsilon(c) \circ F(g)$を満たす。

  この命題より定まる$\Obj_{\bm{J}}$上の関数を$\eta$とする。

  \vskip\baselineskip

  任意の対象$j, j^\prime \in \Obj_{\bm{J}}$と、$\bm{J}$の任意の射$i \colon j \to j^\prime$について、以下が成り立つ。
  \eqa*{
    \epsilon(c) \circ F(\eta(j^\prime) \circ X(i))
    &= \epsilon(c) \circ F(\eta(j^\prime)) \circ \qty(F \circ X)(i) \\*
    &= \theta(j^\prime) \circ \qty(F \circ X)(i) \\*
    &= \theta(j) \\*
    &= \epsilon(c) \circ F(\eta(j))
  }

  一意性より、$\eta(j^\prime) \circ X(i) = \eta(j)$である。

  よって、$\eta$は$X$から$\Delta_{\bm{D}}(G(c))$への自然変換である。

  \vskip\baselineskip

  $x, u$は$X$から$\Delta_{\bm{D}}$への余普遍射であるので、
  $\bm{D}$の射$h \colon x \to G(c)$が一意に存在して、$\eta = \Delta_{\bm{D}}(h) \circ u$を満たす。

  ここで、$f \coloneqq \epsilon(c) \circ F(h)$とすると、任意の対象$j \in \Obj_{\bm{J}}$について以下が成り立つ。
  \eqa*{
    f \circ \qty(\Id_F \ast u)(j)
    &= \epsilon(c) \circ F(h) \circ \qty(\Id_F \ast u)(j) \\*
    &= \epsilon(c) \circ F(h \circ u(j)) \\*
    &= \epsilon(c) \circ F\qty(\eta(j)) \\*
    &= \theta(j)
  }

  したがって、$\Delta_{\bm{C}}(f) \circ \qty(\Id_F \ast u) = \theta$である。

  \vskip\baselineskip

  次に、$\bm{C}$の射$f^\prime \colon F(x) \to c$が存在して、$\Delta_{\bm{C}}(f^\prime) \circ \qty(\Id_F \ast u) = \theta$であるとする。

  $f^\prime = \epsilon(c) \circ F(h^\prime)$を満たす$\bm{D}$の射$h^\prime \colon x \to G(c)$が一意に存在する。

  任意の対象$j \in \Obj_{\bm{J}}$について、以下が成り立つ。
  \eqa*{
    \epsilon(c) \circ F\qty(h \circ u(j))
    &= \theta(j) \\*
    &= f^\prime \circ \qty(\Id_F \ast u)(j) \\*
    &= \epsilon(c) \circ F(h^\prime) \circ \qty(\Id_F \ast u)(j) \\*
    &= \epsilon(c) \circ F\qty(h^\prime \circ u(j))
  }

  再び一意性より、$h \circ u(j) = h^\prime \circ u(j)$である。

  ゆえに、$\Delta_{\bm{D}}(h^\prime) \circ u = \Delta_{\bm{D}}(h) \circ u$である。

  $x, u$は$X$の余極限であるため、普遍性より$h = h^\prime$である。
  したがって、$f = f^\prime$である。
}
