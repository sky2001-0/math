\lsection{写像}

\lsubsection{写像と集合の圏}

\dfn{写像}{
  集合$X, Y$と以下を満たす集合$G \subset X \times Y$を考える。
  \eq*{
    \forall x \in X \exists! y \in Y \qty(\qty(x, y) \in G)
  }

  このとき、順序対$\qty(\qty(X, Y), G)$を写像と呼ぶ。

  写像$f \coloneqq \qty(\qty(X, Y), G)$を、$f \colon X \to Y$で表す。

  写像$f$から定まる$X$上の関数を、記号の濫用であるが、$f$で表す。
}

\rem{関数による写像の定義}{
  集合$X, Y$について、以下を満たす$X$上の関数$f$を考える。
  \eq*{
    \forall x \in X \qty(f(x) \in Y)
  }

  このとき、$\qty(\qty(X, Y), \qty{\qty(x, f(x)) \mid x \in X})$は写像である。

  記号の濫用であるが、この写像を$f$で表す。

  写像の定義$f \coloneqq \qty(\qty(X, Y), \qty{\qty(x, f(x)) \mid x \in X})$を、簡潔に以下でも表す。
  \eq*{
    f \colon X \to Y, x \mapsto f(x)
  }
}

\dfn{恒等写像}{
  集合$X$について、以下で定める$X$上の関数$\id_X$を考える。
  \eq*{
    \id_X(x) \coloneqq x
  }

  このとき、写像$\id_X \colon X \to X$を$X$の恒等写像と呼ぶ。
}

\cor*{
  集合$X, Y$について、以下が成り立つ。
  \eq*{
    X = Y \leftrightarrow \id_X = \id_Y
  }
}

\dfn{写像の合成}{
  集合$X, Y, Z$と、写像$f \colon X \to Y, g \colon Y \to Z$を考える。

  以下で定める$X$上の関数$h$を考える。
  \eq*{
    h(x) \coloneqq g(f(x))
  }

  このとき、写像$h \colon X \to Z$を、$f$と$g$の合成写像、または単に合成と呼び、$g \circ f$で表す。
}

\cor{写像の合成の結合法則}{
  集合$X, Y, Z, W$と、写像$f \colon X \to Y, g \colon Y \to Z, h \colon Z \to W$について、
  以下が成り立つ。
  \eq*{
    \qty(h \circ g) \circ f = h \circ \qty(g \circ f)
  }
}

\rem{射としての写像}{
  写像について、$=, \dom, \cod, \Comp$を以下で定めることにより、\subsecref{射の公理}を満たす。

  集合$X, Y, Z, X^\prime, Y^\prime$と、写像$f \colon X \to Y, g \colon Y \to Z, h \colon X^\prime \to Y^\prime$について、
  \begin{enumerate}
    \item 等号は集合の相等として定める。
    \item $\dom(f) \coloneqq \id_X$
    \item $\cod(f) \coloneqq \id_Y$
    \item $\circ$は\dfnref{写像の合成}の意味として、$\Comp(f, g, h) \defiff h = g \circ f$
  \end{enumerate}
}

\dfn{集合の圏}{
  以下で定める類$A$を考える。
  \eq*{
    f \in A \defiff \text{$f$は写像}
  }

  このとき$A$は、圏である。
  この$A$を、集合の圏と呼び、$\mathbf{Set}$で表す。
}

\dfn{写像全体の集合}{
  集合$X, Y$について、以下で定める集合$Y^X$を、$X$から$Y$への写像全体の集合と呼ぶ。
  \eq*{
    Y^X \coloneqq \Hom_{\mathbf{Set}}\qty(\id_X, \id_Y)
  }

  記号の濫用であるが、$\Hom_{\mathbf{Set}}\qty(X, Y)$と書いて、$Y^X$を表すものとする。
}

\cor*{
  以下が成り立つ。
  \eq*{
    \forall X \qty(X \neq \varnothing \rightarrow \Hom_{\mathbf{Set}}(X, \varnothing) = \varnothing)
  }
}

\dfn{小さい圏}{
  圏$\bm{C}$について、$\bm{C}$の射の上の離散圏$\hat{\bm{C}}$が集合であるとき、$\bm{C}$を小さいという。
}

\cor*{
  小さい圏$\bm{J}$について、$\Obj_{\bm{J}}$は小さい。
}

\dfn{集合から定まる離散圏}{
  集合$X$について、以下で定める類$A$を考える。
  \eq*{
    \id_x \in A \defiff x \in X
  }

  このとき$A$は、小さい離散圏である。
  この$A$を、記号の濫用であるが、$X$で表す。
}


\lsubsection{像と原像、単射と全射}

\dfn{包含写像}{
  集合$X$と$X$の部分$A$について、以下で定める写像$\iota_{A, X} \colon A \to X$を包含写像と呼ぶ。
  \eq*{
    \iota_{A, X}(x) \coloneqq x
  }
}

\dfn{像}{
  写像$f \colon X \to Y$と$X$の部分$A$について、以下で定める集合$f(A)$を$A$の$f$による像と呼ぶ。
  \eq*{
    f(A) \coloneqq \qty{f(x) \mid x \in A}
  }

  また、始域の像$f(X)$を$\Im(f)$で表す。
}

\cor*{
  写像$f \colon X \to Y$について、以下が成り立つ。
  \eq*{
    \forall A_1, A_2 \subset X \qty(A_1 \subset A_2 \rightarrow f(A_1) \subset f(A_2))
  }
}

\dfn{終域の制限}{
  写像$f \colon X \to Y$と$Y$の部分$B$について、$\Im(f) \subset B$とする。

  このとき、写像$\bar{f} \colon X \to B, x \mapsto f(x)$が構成できて、$f = \iota_{B, Y} \circ \bar{f}$が成り立つ。

  この$\bar{f}$を、$f$の終域$B$への制限と呼ぶ。
}

\dfn{原像}{
  写像$f \colon X \to Y$と$Y$の部分$B$について、以下で定める集合$f^{-1}(B)$を$B$の原像と呼ぶ。
  \eq*{
    f^{-1}(B) \coloneqq \qty{x \in X \mid f(x) \in B}
  }
}

\cor*{
  写像$f \colon X \to Y$について、以下が成り立つ。
  \eq*{
    \forall B_1, B_2 \subset Y \qty(B_1 \subset B_2 \rightarrow f^{-1}(B_1) \subset f^{-1}(B_2))
  }
}

\lem{原像は引き戻し}{
  写像$f \colon X \to Y$と、集合$B \subset Y$について、$f \circ \iota_{f^{-1}(B), X}$の終域$B$への制限$\bar{f}$を考える。

  $f^{-1}(B), \qty(\hat{f}, \iota_{f^{-1}(B), X})$は、$\iota_{B, Y}, f$の引き戻しである。
}{
  $\iota_{B, Y} \circ \bar{f} = f \circ \iota_{f^{-1}(B), X}$である。

  集合$Z$と、写像$u_0 \colon Z \to B, u_1 \colon Z \to X$について、$\iota_{B, Y} \circ u_0 = f \circ u_1$とする。

  $\Im(\iota_{B, Y} \circ u_0) \subset B$より、$\Im(u_1) \subset f^{-1}(B)$である。

  ゆえに$u_1$の終域$f^{-1}(B)$への制限$\bar{u}_1$が存在して、$u_1 = \iota_{f^{-1}(B), X} \circ \bar{u}_1$である。

  定義より、$u_1 = \iota_{f^{-1}(B), X} \circ \bar{u}_1$である。

  $z \in Z$について、$\qty(\bar{f} \circ \bar{u}_1)(z) = f(u_1(z)) = u_0(z)$が成り立つ。

  ゆえに、$u_0 = \bar{f} \circ \bar{u}_1$である。

  $\bar{u}_1^\prime \colon Z \to f^{-1}(B)$が存在して、$u_1 = \iota_{f^{-1}(B), X} \circ \bar{u}_1^\prime \land u_0 = \bar{f} \circ \bar{u}_1^\prime$とする。

  $u_1 = \iota_{f^{-1}(B), X} \circ \bar{u}_1^\prime$より、$\bar{u}_1^\prime = \bar{u}_1$である。
}

\dfn{単射}{
  写像$f \colon X \to Y$が以下を満たすとき、$f$を単射という。
  \eq*{
    \forall x, x^\prime \in X \qty(f(x) = f(x^\prime) \rightarrow x = x^\prime)
  }
}

\lem{単射}{
  写像$f \colon X \to Y$について、以下の\num{3}つは同値である。
  \begin{enumerate}
    \item $f$が左簡約可能
    \item $f$は単射
    \item $f$が、空写像または左可逆
  \end{enumerate}
}{
  $1. \rightarrow 2.$を示す。

  $x \in X$について、写像$g_x \colon \qty{\varnothing} \to \dom(f), g_x(\varnothing) \coloneqq x$を考える。

  $f(x) = f(y)$であるとき、$f \circ g_x = f \circ g_y$であり、左簡約可能より$g_x = g_y$すなわち$x = y$である。

  \vskip\baselineskip

  $2. \rightarrow 3.$について、$2.$かつ空写像でないならば左可逆であることを示す。

  空でないので$\exists x_0 \in X$より、以下で定める集合$G$が存在する。
  \eq*{
    G \coloneqq \qty{\qty(f(x), x) \mid x \in X} \cup \qty{\qty(y, x_0) \mid y \in Y \setminus \Im(f)}
  }

  $\qty(\qty(Y, X), G)$は写像であり、定義より$f$の左逆写像となる。

  \vskip\baselineskip

  $3. \rightarrow 1.$を示す。

  $f$が空写像であるときを考える。

  $f$に右から合成可能、すなわち、終域を空集合とする写像は、$g_0 \colon \varnothing \to \varnothing$のみであるので、左簡約可能である。

  $f$が左可逆であるとき、明らか。
}

\cor*{
  包含写像は単射である。
}

\lem*{
  集合$X, Y$について、単射$f \colon X \to Y$が存在するならば、単射$g \colon \P(X) \to \P(Y)$が存在する。
}{
  $g(A) \coloneqq \qty{f(x) \mid x \in A}$で定義する写像は、単射である。
}

\thm{Cantorの対角線論法}{
  単射$f \colon \P(A) \to A$は存在しない。
}{
  存在すると仮定する。

  以下の集合$Y$を考える。
  \eq*{
    Y \coloneqq \qty{f(X) \mid X \subset A \land f(X) \notin X}
  }

  このとき$Y \subset A$である。

  $f(Y) \notin Y$とすると、定義より$f(Y) \in Y$である。
  ゆえに矛盾。

  $f(Y) \in Y$とすると、$\exists Z \subset A \qty(f(Z) \notin Z \land f(Z) = f(Y))$であるが、$f$の単射性より$Y = Z$ゆえに$f(Y) \notin Y$である。
  ゆえに矛盾。

  背理法より示される。
}

\dfn{全射}{
  写像$f \colon X \to Y$が以下を満たすとき、$f$を全射という。
  \eq*{
    \forall y \in Y \exists x \in X \qty(f(x) = y)
  }
}

\lem{全射}{
  写像$f$について、以下の\num{3}つは同値である。
  \begin{enumerate}
    \item $f$が右簡約可能
    \item $f$が全射
    \item $f$が右可逆
  \end{enumerate}
}{
  $1. \rightarrow 2.$を示す。

  全射でないとすると、$\exists y_0 \in Y \forall x \in X \qty(f(x) \neq y_0)$である。

  以下を満たす写像$g, h \in \Hom_{\mathbf{Set}}\qty(Y, \qty{\varnothing, \qty{\varnothing}})$を考える。
  \eq*{
    g(y) \coloneqq \varnothing
    \qquad
    h(y) \coloneqq
    \begin{cases}
      \qty{\varnothing} & \qty(y = y_0) \\
      \varnothing & \qty(y \neq y_0)
    \end{cases}
  }

  $g \circ f = h \circ f$であるが、$g \neq h$であり、右簡約可能ではない。

  対偶法より示される。

  \vskip\baselineskip

  $2. \rightarrow 3.$を示す。

  \axiref{Skolem関係の導入}の与える写像は、右逆写像である。

  \vskip\baselineskip

  $3. \rightarrow 1.$は明らか。
}

\cor*{
  写像$f$の終域$\Im(f)$への制限は全射である。
}

\lem{像の普遍性}{
  写像$f \colon X \to Y$と、$f$の終域$\Im(f)$への制限$\bar{f}$について、
  写像$g \colon X \to Z$と単射$m \colon Z \to Y$が以下を満たすとする。
  \eq*{
    f = m \circ g
  }

  このとき、写像$h \colon \Im(f) \to Z$が一意に存在して、$\iota_{\Im(f), Y} = m \circ h$である。

  さらに、$h$は単射である。

  また、$g$が全射ならば、$h$は全単射となる。
}{
  以下も命題を示す。
  \eq*{
    \forall y \in \Im(f) \exists! z \in Z \qty(y = m(z))
  }

  任意の$y \in \Im(f)$について、$\bar{f}$は全射より、$x \in X$が存在して$y = f(x)$である。

  $z = g(x)$について、$y = f(x) = \qty(m \circ g)(x) = m(z)$である。

  $m$は単射であるから、このような$z$は一意である。

  この命題が定める写像を$h \colon \Im(f) \to Z$とする。

  \vskip\baselineskip

  $h$の定義より、任意の$y \in \Im(f)$について$m(h(y)) = y = \iota_{\Im(f), Y}(y)$が成り立つ。

  したがって、$\iota_{\Im(f), Y} = m \circ h$である。

  $m \circ h$が左簡約可能より、$h$は左簡約可能すなわち単射である。

  \vskip\baselineskip

  写像$h^\prime \colon \Im(f) \to Z$が$\iota_{\Im(f), Y} = m \circ h^\prime$を満たすとする。

  $m \circ h = m \circ h^\prime(y)$より、$m$は単射であるので、$h = h^\prime$である。

  \vskip\baselineskip

  また、$m \circ h \circ \bar{f} = m \circ g$より、$m$が単射であるので、$h \circ \bar{f} = g$である。

  $g$が全射であるとする。

  $h \circ \bar{f}$が右簡約可能より、$h$は右簡約可能すなわち全射である。
}

\dfn{全単射}{
  写像$f$が同型射であるとき、$f$を全単射と呼ぶ。
}

\dfn{逆写像}{
  写像$f$の逆射を逆写像と呼ぶ。
}

\cor*{
  全単射$f \colon X \to Y$と$Y$の部分$B$について、原像$f^{-1}(B)$と逆写像の像$f^{-1}(B)$は一致する。
}


\lsubsection{集合の圏の普遍性}

\cor*{
  任意の単集合は$\mathbf{Set}$の終対象である。
}

\cor*{
  空集合は$\mathbf{Set}$の始対象である。
}

\dfn{空写像}{
  空集合は$\mathbf{Set}$の始対象であるので、終域$Y$について、始域が空集合である写像は一意に定まる。

  この写像$\varnothing \to Y$を、$Y$の空写像と呼ぶ。
}

\lem{集合の圏における等化子の存在}{
  写像$f, g \colon X \to Y$を考える。

  以下で定める$\Eq(f, g), \pi$は、$f, g$の等化子である。
  \eqg*{
    \Eq(f, g) \coloneqq \qty{x \in X \mid f(x) = g(x)} \\*
    \pi \coloneqq \iota_{\Eq(f, g), X}
  }
}{
  定義より$f \circ \pi = g \circ \pi$である。

  \vskip\baselineskip

  $f \circ h = g \circ h$なる写像$h \colon Z \to X$を考える。

  $\forall z \in Z \qty(f(h(z)) = g(h(z)))$より、$\Im(h) \subset \Eq(f, g)$である。

  ゆえに$h$の終域$\Eq(f, g)$への制限$\bar{h}$が存在して、$h = \pi \circ \bar{h}$である。

  写像$\bar{h}^\prime \colon Z \to \Eq(f, g)$であって、$h = \pi \circ \bar{h}^\prime$とする。

  任意の$z \in Z$について、以下が成り立つ。
  \eq*{
    \bar{h}^\prime(z) = \pi(\bar{h}^\prime(z)) = h(z) = \pi(\bar{h}(z)) = \bar{h}(z)
  }

  したがって一意である。
}

\lem{集合の圏における小さい積の存在}{
  小さい離散圏$\bm{J}$と、関手$X \colon \bm{J} \to \mathbf{Set}$を考える。

  以下で定める$\prod X, \pi$は$X$の積である。
  \eqg*{
    \prod X \coloneqq \qty{f \in \Hom_{\mathbf{Set}}\qty(\bm{J}, \bigcup \qty{X(j) \mid j \in \bm{J}}) \mid \forall j \in \bm{J} \qty(f(j) \in X(j))} \\*
    \pi(j) \colon \prod X \to X(j), f \mapsto f(j)
  }
}{
  $\bm{J}$が離散圏であるので、$\pi$は$\Delta(\prod X)$から$\Id_X$への自然変換である。

  対角関手$\Delta \colon \mathbf{Set} \to \Func(\bm{J}, \mathbf{Set})$を考える。

  \vskip\baselineskip

  集合$A$と、自然変換$u \colon \Delta(A) \to \Id_X$について、以下で定める写像$h \colon A \to \prod X$を考える。
  \eq*{
    h(a) \colon \bm{J} \to \bigcup \qty{X(j) \mid j \in \bm{J}}, j \mapsto u(j)(a)
  }

  定義より、$u = \pi \circ \Delta(h)$である。

  \vskip\baselineskip

  写像$h^\prime \colon A \to \prod X$が、$u = \pi \circ \Delta(h^\prime)$を満たすとする。

  任意の要素$a \in A$と、任意の対象$j \in \bm{J}$について、以下を満たす。
  \eqa*{
    h^\prime(a)(j)
    &= \pi(j)(h^\prime(a)) \\*
    &= \qty(\pi(j) \circ h^\prime)(a) \\*
    &= u(j)(a) \\*
    &= h(a)(j)
  }

  したがって、$h^\prime = h$である。
}

\thm{集合の圏における小さい極限の存在}{
  小さい圏$\bm{J}$と、関手$X \colon \bm{J} \to \mathbf{Set}$について、$X$の極限が存在する。
}{
  \lemref{集合の圏における等化子の存在}、\lemref{集合の圏における小さい積の存在}、\thmref{極限の存在定理}より、成り立つ。
}

\cor*{
  集合$X, Y$と、以下で定める写像$\pi_0, \pi_1$を考える。
  \eq*{
    \pi_0 \colon X \times Y \to X, \qty(x, y) \mapsto x
    \qquad
    \pi_1 \colon X \times Y \to Y, \qty(x, y) \mapsto y
  }

  このとき、$X \times Y, \qty(\pi_0, \pi_1)$は$X, Y$の二項積である。
}

\thm{選択関数の存在}{
  任意の集合$X$について、写像$f \colon X \setminus \qty{\varnothing} \to \bigcup X$が存在して、以下が成り立つ。
  \eq*{
    \forall x \in X \setminus \qty{\varnothing} \qty(f(x) \in x)
  }
}{
  空集合の定義より、以下の命題が成り立つ。
  \eq*{
    \forall x \in X \setminus \qty{\varnothing} \exists y \qty(y \in x)
  }

  \axiref{Skolem関係の導入}より、$X \setminus \qty{\varnothing}$上の関数$f$が存在する。

  任意の$x \in X \setminus \qty{\varnothing}$について、$f(x) \in x$より$f(x) \in \bigcup X$である。

  ゆえに、$f \colon X \setminus \qty{\varnothing} \to \bigcup X$が与えられる。
}

\lem{非空からなる積は非空}{
  空でない集合$J$から$\mathbf{Set}$への関手$X$について、以下が成り立つ。
  \eq*{
    \forall j \in J \qty(X(j) \neq \varnothing) \rightarrow \prod X \neq \varnothing
  }
}{
  $Z \coloneqq \qty{X(j) \mid j \in J}$について、定義より$Z \neq \varnothing \land \varnothing \notin Z$である。

  \thmref{選択関数の存在}より、写像$f \colon Z \to \bigcup Z$が存在して、$\forall z \in Z \qty(f(z) \in z)$である。

  写像$g \colon J \to \bigcup Z, j \mapsto f(X(j))$で定めると、$\forall j \in J \qty(g(j) \in X(j))$であるので、$g \in \prod X$である。

  ゆえに、$\prod X \neq \varnothing$である。
}

\dfn{商写像}{
  集合$Y$と、$Y$の分割$P$について、定義より以下が成り立つ。
  \eq*{
    \forall y \in Y \exists! A \in P \qty(y \in A)
  }

  この命題が定める写像$Y \to P$を、分割$P$の商写像と呼ぶ。
}

\lem{集合の圏における余等化子の存在}{
  写像$f, g \colon X \to Y$について、以下で定める集合$P$は$Y$の分割である。
  \eq*{
    P \coloneqq \qty{\bigcap \qty{A \subset Y \mid y \in A \land \forall x \in X \qty(f(x) \in A \leftrightarrow g(x) \in A)} \mid y \in Y}
  }

  さらに、$P$の商写像$q \colon Y \to P$について、$P, q$は余等化子である。
}{
  要素$y \in Y$について、集合$B_y \coloneqq \qty{A \subset Y \mid y \in A \land \forall x \in X \qty(f(x) \in A \leftrightarrow g(x) \in A)}$を考える。

  $Y \in B_y$より$B_y \neq \varnothing$であるので、$y \in \bigcap B_y$である。

  定義より$\bigcap B_y \in B_y$であり、$\bigcap B_y \subset Y$である。

  $P = \qty{\bigcap B_y \mid y \in Y}$であることに注意して、$Y = \bigcup P$である。

  \vskip\baselineskip

  $y_1, y_2 \in Y$について、$\exists y \in Y \qty(y \in \qty(\bigcap B_{y_1}) \cap \qty(\bigcap B_{y_2}))$とする。

  任意の$A \in B_{y_1}$について、$y \in \bigcap B_{y_1} \subset A$より、$A \in B_y$である。
  よって、$B_{y_1} \subset B_y$である。

  $A^\prime \in B_y$について、$y_1 \notin A^\prime$とすると、$y_1 \in Y \setminus A^\prime$より、$Y \setminus A^\prime \in B_{y_1}$である。

  $B_{y_1} \subset B_y$より、$Y \setminus A^\prime \in B_y$であるが、これは$y \in A^\prime$に反する。

  よって$y_1 \in A^\prime$である。

  したがって、$B_y \subset B_{y_1}$である。

  ゆえに$B_{y_1} = B_y$である。

  同様にして、$B_{y_2} = B_y$である。

  $\qty(\bigcap B_{y_1}) \cap \qty(\bigcap B_{y_2}) \rightarrow \bigcap B_{y_1} = \bigcap B_{y_2}$であるので、$P$は$Y$の分割である。

  \vskip\baselineskip

  任意の要素$x \in X$を考える。

  $\bigcap B_{f(x)} \in B_{f(x)}$より$g(x) \in \bigcap B_{f(x)}$であるので、
  $\bigcap B_{f(x)} \in B_{g(x)}$であり、$\bigcap B_{g(x)} \subset \bigcap B_{f(x)}$が成り立つ。

  同様に$\bigcap B_{f(x)} \subset \bigcap B_{g(x)}$であるので、$\bigcap B_{f(x)} = \bigcap B_{g(x)}$である。

  ゆえに、$\qty(q \circ f)(x) = \bigcap B_{f(x)} = \bigcap B_{g(x)} = \qty(q \circ g)(x)$が成り立つ。

  したがって、$q \circ f = q \circ g$である。

  \vskip\baselineskip

  写像$h \colon Y \to Z$が$h \circ f = h \circ g$を満たすとする。

  任意の要素$y \in Y$について、$y \in h^{-1}\qty(\qty{h(y)})$である。

  任意の要素$x \in X$について、$f(x) \in h^{-1}\qty(\qty{h(y)})$ならば$h(y) = h(f(x)) = h(g(x))$より$g(x) \in h^{-1}\qty(\qty{h(y)})$である。

  同様に、$g(x) \in h^{-1}\qty(\qty{h(y)})$ならば$f(x) \in h^{-1}\qty(\qty{h(y)})$である。

  したがって、$h^{-1}\qty(\qty{h(y)}) \in B_y$であり、$\bigcap B_y \subset h^{-1}\qty(\qty{h(y)})$である。

  任意の要素$y_1, y_2 \in Y$が$\bigcap B_{y_1} = \bigcap B_{y_2}$であるならば、$y_1, y_2 \in h^{-1}\qty(\qty{h(y_1)})$から$h(y_1) = h(y_2)$を満たす。

  したがって、$k \coloneqq \qty(\qty(P, Z), \qty{\qty(\bigcap B_y, h(y)) \mid y \in Y})$は写像である。

  任意の$y \in Y$について$(k \circ q)(y) = k\qty(\bigcap B_y) = h(y)$であるから、$k \circ q = h$が成り立つ。

  \vskip\baselineskip

  一意性を示す。

  写像$k^\prime \colon P \to Z$が$k^\prime \circ q = h$を満たすとする。

  任意の$p \in P$について、定義よりある$y \in Y$が存在して$p = \bigcap B_y$であるので、以下が成り立つ。
  \eq*{
    k^\prime(p) = k^\prime\qty(\bigcap B_y) = k^\prime(q(y)) = h(y) = k(q(y)) = k\qty(\bigcap B_y) = k(p)
  }

  したがって、$k^\prime = k$である。
}

\lem{集合の圏における小さい和の存在}{
  小さい離散圏$\bm{J}$と、$\bm{J}$から$\mathbf{Set}$への関手$X$を考える。

  このとき、以下で定める$\coprod X, \kappa$は$X$の和である。
  \eqg*{
    \coprod X \coloneqq \bigcup \qty{\qty{\qty(j, x) \mid x \in X(j)} \mid j \in \bm{J}} \\*
    \kappa(j) \colon X(j) \to \coprod X, x \mapsto \qty(j, x)
  }
}{
  対角関手$\Delta \colon \mathbf{Set} \to \Func(\bm{J}, \mathbf{Set})$を考える。

  $\bm{J}$が離散圏であるので、$\kappa$は$\Id_X$から$\Delta(\coprod X)$への自然変換である。

  \vskip\baselineskip

  集合$A$と、自然変換$u \colon \Id_X \to \Delta(A)$を考える。

  写像$h \colon \coprod X \to A, \qty(j, x) \mapsto u(j)(x)$を考える。

  任意の要素$x \in X(j)$と任意の射$j \in \bm{J}$について、以下が成り立つ。
  \eqa*{
    \qty(h \circ \kappa(j))(x)
    &= h\qty(\kappa(j)(x)) \\*
    &= h\qty(\qty(j, x)) \\*
    &= u(j)(x)
  }

  ゆえに、$u = \Delta(h) \circ \kappa$が成り立つ。

  \vskip\baselineskip

  写像$h^\prime \colon \coprod X \to A$が$u = \Delta(h^\prime) \circ \kappa$を満たすとする。

  任意の$\qty(j, x) \in \coprod X$について、以下を満たす。
  \eqa*{
    h^\prime\qty(\qty(j, x))
    &= h^\prime(\kappa(j)(x)) \\*
    &= \qty(\Delta(h^\prime)(j) \circ \kappa(j))(x) \\*
    &= u(j)(x) \\*
    &= h\qty(\qty(j, x)) \\*
  }

  したがって、$h^\prime = h$である。
}

  \begin{enumerate}

    \item $\dom(\qty(f_1, g_1)) \coloneqq \qty(\dom(f_1), \dom(g_1))$
    \item $\cod(\qty(f_1, g_1)) \coloneqq \qty(\cod(f_1), \cod(g_1))$
